33bool UBaseSkillAbility::CalculateFinalStatus()
34{
35 if (!Attributes) return false;
36
37 StackedStatusData.AttackArea = 0;
38 StackedStatusData.CooldownTime = 0;
39 StackedStatusData.MaxPierceCount = 0;
40 StackedStatusData.NumToSpawn = 0;
41 StackedStatusData.ProjectileSizeMultiplier = 0;
42 StackedStatusData.ProjectileSpeed = 0;
43 StackedStatusData.SkillDamage = 0;
44 StackedStatusData.ProjectileLife = 0;
45
46 CalculatedStatusData.NumToSpawn = SkillDataTable.NumToSpawn + Attributes->GetAdditionalProjectileCount();
47 CalculatedStatusData.ProjectileSpeed = SkillDataTable.ProjectileSpeed * Attributes->GetProjectileSpeedMultiplier();
48 CalculatedStatusData.ProjectileSizeMultiplier = SkillDataTable.ProjectileSizeMultiplier * Attributes->GetProjectileSizeMultiplier();
49 CalculatedStatusData.CooldownTime = SkillDataTable.CooldownTime * Attributes->GetPassiveSkillCooldownReduction();
50 CalculatedStatusData.AttackArea = SkillDataTable.AttackArea * Attributes->GetAttackAreaMultiplier();
51 CalculatedStatusData.SkillDamage = SkillDataTable.SkillDamage * Attributes->GetAttackPower();
52 CalculatedStatusData.MaxPierceCount = SkillDataTable.MaxPierceCount;
53 CalculatedStatusData.ProjectileLife = SkillDataTable.ProjectileLife;
54
55 UAbilitySystemComponent* ASC = GetAbilitySystemComponentFromActorInfo();
56 if (!ASC) return false;
57
58 FGameplayTagContainer QueryTags;
59 QueryTags.AddTag(SkillTag);
60
61 TArray<FActiveGameplayEffectHandle> FoundEffects = ASC->GetActiveEffectsWithAllTags(QueryTags);
62
63
64 for (auto& Handle : FoundEffects)
65 {
66 const FGameplayEffectSpec& Spec = ASC->GetActiveGameplayEffect(Handle)->Spec;
67 float tmp = 0;
68
69 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.NumToSpawn")), false, -100.f);
70 if (tmp != -100)
71 {
72 StackedStatusData.NumToSpawn += tmp;
73 }
74
75 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.ProjectileSpeed")), false, -100.f);
76 if (tmp != -100)
77 {
78 StackedStatusData.ProjectileSpeed += tmp;
79 }
80
81 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.ProjectileSize")), false, -100.f);
82 if (tmp != -100)
83 {
84 StackedStatusData.ProjectileSizeMultiplier += tmp;
85 }
86
87 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.Cooldown")), false, -100.f);
88 if (tmp != -100)
89 {
90 StackedStatusData.CooldownTime += tmp;
91 }
92
93 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.AttackArea")), false, -100.f);
94 if (tmp != -100)
95 {
96 StackedStatusData.AttackArea += tmp;
97 }
98
99 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.Damage")), false, -100.f);
100 if (tmp != -100)
101 {
102 StackedStatusData.SkillDamage += tmp;
103 }
104
105 tmp = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.Pierce")), false, -100.f);
106 if (tmp != -100)
107 {
108 if (tmp == -1 || CalculatedStatusData.MaxPierceCount == -1) StackedStatusData.MaxPierceCount = -1;
109 else StackedStatusData.MaxPierceCount += tmp;
110 }
111
112 }
113 CalculatedStatusData.NumToSpawn += StackedStatusData.NumToSpawn;
114 CalculatedStatusData.ProjectileSpeed *= FMath::Max(0.f,1.f + StackedStatusData.ProjectileSpeed);
115 CalculatedStatusData.ProjectileSizeMultiplier *= FMath::Max(0.f, 1.f + StackedStatusData.ProjectileSizeMultiplier);
116 CalculatedStatusData.CooldownTime *= FMath::Max(0.f,1.f + StackedStatusData.CooldownTime);
117 CalculatedStatusData.AttackArea *= FMath::Max(0.f,1.f + StackedStatusData.AttackArea);
118 CalculatedStatusData.SkillDamage *= FMath::Max(0.f,1.f + StackedStatusData.SkillDamage);
119 if (CalculatedStatusData.MaxPierceCount == -1 || StackedStatusData.MaxPierceCount == -1)
120 CalculatedStatusData.MaxPierceCount = -1;
121 else CalculatedStatusData.MaxPierceCount += StackedStatusData.MaxPierceCount;
122 return true;
123
124}

166void UBaseSkillAbility::SpawnAndInitializeProjectile(const FTransform& SpawnTransform, int32 Index, int32 TotalCount)
167{
168
169 if (!GetWorld()->GetAuthGameMode() || !BaseSkillData.ProjectileClass)
170 {
171 return;
172 }
173
174 UProjectilePoolSubsystem* PoolSubsystem = GetWorld()->GetSubsystem<UProjectilePoolSubsystem>();
175 if (!PoolSubsystem)
176 {
177 UE_LOG(LogTemp, Error, TEXT("UProjectilePoolSubsystem not found"));
178 return;
179 }
180
181 AProjectileBase* Projectile = PoolSubsystem->RequestProjectile(TSubclassOf<AProjectileBase>(BaseSkillData.ProjectileClass));
182 if (!Projectile)
183 {
184 return;
185 }
186 Projectile->PayloadEffectClass = BaseSkillData.PayloadEffectClass;
187
188 if (BaseSkillData.MovementStrategyClass)
189 {
190 UMovementStrategyBase* MovementStrategy = Projectile->FindComponentByClass<UMovementStrategyBase>();
191
192 // 1. 기존 컴포넌트가 있지만, 원하는 클래스가 아닌 경우
193 if (MovementStrategy && MovementStrategy->GetClass() != BaseSkillData.MovementStrategyClass)
194 {
195 MovementStrategy->DestroyComponent();
196 MovementStrategy = nullptr;
197 }
198
199 // 2. 컴포넌트가 아예 없거나, 잘못된 것을 방금 파괴한 경우
200 if (!MovementStrategy)
201 {
202 MovementStrategy = NewObject<UMovementStrategyBase>(Projectile, BaseSkillData.MovementStrategyClass);
203 if (MovementStrategy)
204 {
205 MovementStrategy->RegisterComponent();
206 }
207 }
208
209 }
210
211 Projectile->ActivateProjectile(
212 CalculatedStatusData,
213 VFXCueTag,
214 GetAbilitySystemComponentFromActorInfo(),
215 SpawnTransform,
216 GetAbilitySystemComponentFromActorInfo()->GetAvatarActor(),
217 Cast<APawn>(GetAvatarActorFromActorInfo()),
218 Index, TotalCount
219 );
220
221
222
223}
