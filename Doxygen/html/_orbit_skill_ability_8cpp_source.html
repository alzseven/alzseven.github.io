21TArray<FTransform> UOrbitSkillAbility::CalculateSpawnTransforms_Implementation(AActor* Caster, int32 SpawnCount)
22{
23 TArray<FTransform> Transforms;
24
25 // 유효성 검사: Caster가 없거나 생성할 개수가 0 이하면 빈 배열을 반환합니다.
26 if (!Caster || SpawnCount <= 0)
27 {
28 return Transforms;
29 }
30
31
32 const FVector CasterLocation = Caster->GetActorLocation();
33
34 // 이 값은 데이터 테이블(DT_SkillLevelData)에서 읽어온 'EffectRadius'가 되어야 합니다.
35 // 여기서는 예시로 150.0f를 사용합니다.
36 const float SpawnRadius = CalculatedStatusData.AttackArea;
37
38 // 투사체 사이의 각도 간격을 계산합니다.
39 const float AngleStep = 360.0f / SpawnCount;
40
41 for (int32 i = 0; i < SpawnCount; ++i)
42 {
43 // 현재 투사체의 각도를 계산합니다.
44 const float CurrentAngle = AngleStep * i;
45 const float AngleInRadians = FMath::DegreesToRadians(CurrentAngle);
46
47 // 중심점(Caster)으로부터의 오프셋(거리와 방향)을 계산합니다.
48 // Z축 값은 0으로 하여 Caster와 같은 높이에 스폰되도록 합니다.
49 const FVector Offset = FVector(SpawnRadius * FMath::Cos(AngleInRadians), SpawnRadius * FMath::Sin(AngleInRadians), 0.f);
50
51 // 최종 스폰 위치를 계산합니다.
52 const FVector SpawnLocation = CasterLocation + Offset;
53
54 // 투사체가 원의 바깥쪽을 바라보도록 회전값을 계산합니다.
55 // Offset 벡터의 방향이 바로 바깥쪽을 향하는 방향이므로, 이 벡터를 사용해 회전값을 만듭니다.
56 const FRotator SpawnRotation = Offset.Rotation() + FRotator(0,90,0);
57 // 계산된 위치와 회전값으로 Transform을 만들고 배열에 추가합니다.
58 Transforms.Emplace(SpawnRotation, SpawnLocation);
59 }
60
61 return Transforms;
62}
