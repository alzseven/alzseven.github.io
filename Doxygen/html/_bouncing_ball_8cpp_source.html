49void ABouncingBall::LaunchTowards(const FVector& TargetLocation)
50{
51 if (!HasAuthority()) return;
52
53 ProjectileMovementComp->Activate(true);
54
55 FVector LaunchVelocity;
56 const float GravityZ = GetWorld()->GetGravityZ() * ProjectileMovementComp->ProjectileGravityScale;
57 const bool bSuccess = UGameplayStatics::SuggestProjectileVelocity(
58 this,
59 LaunchVelocity,
60 GetActorLocation(),
61 TargetLocation,
62 700.f,
63 true,
64 0,
65 GravityZ,
66 ESuggestProjVelocityTraceOption::DoNotTrace
67 );
68
69 if (bSuccess)
70 {
71 ProjectileMovementComp->Velocity = LaunchVelocity;
72 }
73 else
74 {
75 const FVector FallbackVelocity = (TargetLocation - GetActorLocation()).GetSafeNormal() * 1500.f;
76 ProjectileMovementComp->Velocity = FallbackVelocity;
77 }
78}
