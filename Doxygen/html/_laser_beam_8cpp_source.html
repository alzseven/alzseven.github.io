163void ALaserBeam::CheckOverlappingActors()
164{
165 CurrentHitActors.Empty();
166
167 if (!LaserCollision) return;
168
169 TArray<AActor*> OverlappingActors;
170 LaserCollision->GetOverlappingActors(OverlappingActors);
171
172 AActor* OwnerActor = GetOwner();
173 for (AActor* Actor : OverlappingActors)
174 {
175 if (!Actor || Actor == this || Actor == OwnerActor) continue;
176
177 if (Cast<IAbilitySystemInterface>(Actor) && Actor->ActorHasTag(FName("Player")))
178 {
179 CurrentHitActors.Add(Actor);
180 }
181 }
182}

184void ALaserBeam::ApplyEffectToTarget(AActor* Target, float DamageAmount)
185{
186 if (!SourceASC || !PayloadEffectClass || !Target) return;
187
188 UAbilitySystemComponent* TargetASC = UAbilitySystemBlueprintLibrary::GetAbilitySystemComponent(Target);
189 if (!TargetASC) return;
190
191 FGameplayEffectContextHandle EffectContext = SourceASC->MakeEffectContext();
192 EffectContext.AddSourceObject(SourceASC->GetOwner());
193 FGameplayEffectSpecHandle SpecHandle = SourceASC->MakeOutgoingSpec(PayloadEffectClass, 1.f, EffectContext);
194 if (SpecHandle.IsValid())
195 {
196 UAbilitySystemBlueprintLibrary::AssignTagSetByCallerMagnitude(SpecHandle, DamageTag, DamageAmount);
197 TargetASC->ApplyGameplayEffectSpecToSelf(*SpecHandle.Data.Get());
198 }
199}
