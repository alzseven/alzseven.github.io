58void UStraightSeqSkillAbility::SpawnAndInitializeProjectile()
59{
60
61 if (!GetWorld()->GetAuthGameMode() ||!BaseSkillData.ProjectileClass)
62 {
63 return;
64 }
65 FTransform SpawnTransform = CalculateSpawnTransform(GetAvatarActorFromActorInfo(), SpawnCount);
66 // 발사 이펙트 GameplayCue 실행
67
68 UProjectilePoolSubsystem* PoolSubsystem = GetWorld()->GetSubsystem<UProjectilePoolSubsystem>();
69 if (!PoolSubsystem)
70 {
71 UE_LOG(LogTemp, Error, TEXT("UProjectilePoolSubsystem not found"));
72 return;
73 }
74
75 AProjectileBase* Projectile = PoolSubsystem->RequestProjectile(TSubclassOf<AProjectileBase>(BaseSkillData.ProjectileClass));
76 if (!Projectile)
77 {
78 return;
79 }
80 Projectile->PayloadEffectClass = BaseSkillData.PayloadEffectClass;
81
82 if (BaseSkillData.MovementStrategyClass)
83 {
84 UMovementStrategyBase* MovementStrategy = Projectile->FindComponentByClass<UMovementStrategyBase>();
85
86 // 1. 기존 컴포넌트가 있지만, 원하는 클래스가 아닌 경우
87 if (MovementStrategy && MovementStrategy->GetClass() != BaseSkillData.MovementStrategyClass)
88 {
89 MovementStrategy->DestroyComponent();
90 MovementStrategy = nullptr;
91 }
92
93 // 2. 컴포넌트가 아예 없거나, 잘못된 것을 방금 파괴한 경우
94 if (!MovementStrategy)
95 {
96 MovementStrategy = NewObject<UMovementStrategyBase>(Projectile, BaseSkillData.MovementStrategyClass);
97 if (MovementStrategy)
98 {
99 MovementStrategy->RegisterComponent();
100 }
101 }
102
103 }
104
105 Projectile->ActivateProjectile(
106 CalculatedStatusData,
107 VFXCueTag,
108 GetAbilitySystemComponentFromActorInfo(),
109 SpawnTransform,
110 GetOwningActorFromActorInfo(),
111 Cast<APawn>(GetAvatarActorFromActorInfo()),
112 0,1
113 );
114
115
116 SpawnCount++;
117 if (SpawnCount >= CalculatedStatusData.NumToSpawn)
118 {
119 ApplySkillCooldown();
120 OnEndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, false);
121
122 }
123
124}
