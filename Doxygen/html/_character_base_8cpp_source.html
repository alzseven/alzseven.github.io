170void ACharacterBase::GrantDefaultAbilities()
171{
172
173 // ASC가 유효한지 확인 (안전장치)
174 if (!AbilitySystemComponent)
175 {
176 UE_LOG(LogTemp, Error, TEXT("!ASC Not Found!"));
177 return;
178 }
179 // DefaultAbilities 배열에 있는 모든 어빌리티를 부여합니다
180 for (TSubclassOf<UGameplayAbility>& AbilityClass : DefaultAbilities)
181 {
182 // 어빌리티 클래스가 유효한지 확인
183 if (AbilityClass)
184 {
185 // 어빌리티 스펙 생성
186 // 파라미터: (어빌리티 클래스, 레벨, 입력 ID, 소스 오브젝트)
187 // INDEX_NONE: 특정 입력에 바인딩하지 않음 (나중에 태그로 활성화)
188 FGameplayAbilitySpec AbilitySpec(AbilityClass, 1, INDEX_NONE, this);
189
190 // ASC에 어빌리티 부여
191 AbilitySystemComponent->GiveAbility(AbilitySpec);
192 UE_LOG(LogTemp, Warning, TEXT("AbilityClass : '%s' Added"), *AbilityClass->GetName());
193 }
194 }
195}

204bool ACharacterBase::IsAlive() const
205{

207 return PS && PS->GetAttributeSet() && PS->GetAttributeSet()->GetHealth() > 0.f;
208}
