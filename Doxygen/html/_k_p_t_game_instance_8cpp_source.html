59void UKPTGameInstance::JoinSessionByIndex(int32 Index)
60{

64 UE_LOG(LogTemp, Error, TEXT("JoinSessionByIndex failed: SessionSubsystem is null."));

70 // 커스텀 구조체 내부의 원본 OnlineResult를 사용해 참여 요청
71 SessionSubsystem->JoinSession(StoredSessionResults[Index].OnlineResult);

73 else
74 {
75 UE_LOG(LogTemp, Warning, TEXT("JoinSessionByIndex failed: Invalid index %d."), Index);

109void UKPTGameInstance::HandleFindSessionsComplete(bool bWasSuccessful, const TArray<FOnlineSessionSearchResult>& SearchResults)
110{
111 // 세션 검색 UI를 닫는 로직을 여기에 추가할 수 있습니다.
112 StoredSessionResults.Empty();
113
114 if (bWasSuccessful)
115 {
116 for (const FOnlineSessionSearchResult& Result : SearchResults)
117 {
118 if (!Result.IsValid()) continue;
119
120 FBlueprintSessionResult BPResult;
121 BPResult.OnlineResult = Result;
122 StoredSessionResults.Add(BPResult);
123 }
124 }
125
126 // 변환된 결과를 블루프린트에 브로드캐스트합니다.
127 OnSessionSearchFinished.Broadcast(StoredSessionResults);
128}
