43void ACargoSpawner::SpawnCarriage(FCargoData InData)
44{
45 if (!CarriageClass)
46 {
47 return;
48 }
49
50 const FVector SpawnLocation = SplineComp->GetLocationAtDistanceAlongSpline(0.0f, ESplineCoordinateSpace::World);
51 const FRotator SpawnRotation = SplineComp->GetRotationAtDistanceAlongSpline(0.0f, ESplineCoordinateSpace::World);
52 SpawnData = InData;
53 FActorSpawnParameters SpawnParams;
54 SpawnParams.Owner = this;
55
56 if (ABaseCargo* NewCargo = GetWorld()->SpawnActor<ABaseCargo>(CarriageClass, SpawnLocation, SpawnRotation, SpawnParams))
57 {
58 NewCargo->InitializeCarriage(SpawnData);
59 NewCargo->SetPath(this);
60 CurrentCarriage = NewCargo;
61 }
62}
