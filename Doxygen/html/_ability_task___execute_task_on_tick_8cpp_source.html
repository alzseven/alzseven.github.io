26void UAbilityTask_ExecuteTaskOnTick::TickTask(float DeltaTime)
27{
28 Super::TickTask(DeltaTime);
29
30 if (!ShouldBroadcastAbilityTaskDelegates())
31 {
32 EndTask();
33 return;
34 }
35 ElapsedTime += DeltaTime;
36 float Alpha = FMath::Clamp(ElapsedTime / TotalDuration, 0.0f, 1.0f);
37
38 if (Curve)
39 {
40 FLinearColor Value = Curve->GetLinearColorValue(Alpha);
41
42 Value.R += FMath::FRandRange(-0.05f, 0.05f);
43 Value.G += FMath::FRandRange(-0.05f, 0.05f);
44 Value.B += FMath::FRandRange(-0.05f, 0.05f);
45
46 if (Alpha < 1.0f)
47 {
48 OnAbilityTaskTick.Broadcast(Alpha, Value);
49 }
50 else
51 {
52 OnAbilityTaskFinish.Broadcast(Alpha, Value);
53 EndTask();
54 }
55 }
