65void UBaseLaserAttackAbility::SpawnAndStartLasers()
66{
67 if (!GetActorInfo().IsNetAuthority() || !LaserBeamClass)
68 {
69 // If called on a client, or if LaserBeamClass is not set, end the ability.
70 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, true);
71 return;
72 }
73
74 // Calculate spawn transforms
75 AvatarInitialLocation = GetAvatarActorFromActorInfo()->GetActorLocation();
76 InitialSpawnTransforms = CalculateLaserSpawnTransforms();
77 SpawnedLasers.Empty();
78
79 UAbilitySystemComponent* SourceASC = GetAbilitySystemComponentFromActorInfo();
80
81 for (const FTransform& SpawnTransform : InitialSpawnTransforms)
82 {
83 // Use SpawnActorDeferred to initialize the laser before it's fully spawned
84 ALaserBeam* Laser = GetWorld()->SpawnActorDeferred<ALaserBeam>(
85 LaserBeamClass,
86 SpawnTransform,
87 GetAvatarActorFromActorInfo(),
88 Cast<APawn>(GetAvatarActorFromActorInfo()),
89 ESpawnActorCollisionHandlingMethod::AlwaysSpawn
90 );
91
92 if (Laser)
93 {
94 Laser->InitiallizeLaser(LaserLength, LaserWidth, DamagePerSecond, PayloadEffectClass, DamageGameplayTag, SourceASC);
95 UGameplayStatics::FinishSpawningActor(Laser, SpawnTransform);
96 Laser->ActivateLaser();
97 SpawnedLasers.Add(Laser);
98 }
99 }
100
101 if (SpawnedLasers.Num() > 0)
102 {
103 // Start the damage tick timer
104 GetWorld()->GetTimerManager().SetTimer(
105 DamageTimerHandle,
106 this,
107 &UBaseLaserAttackAbility::ProcessDamageTick,
108 DamageTickInterval,
109 true // Loop
110 );
111
112 // Start the rotation timer
113 GetWorld()->GetTimerManager().SetTimer(
114 RotationTimerHandle,
115 this,
116 &UBaseLaserAttackAbility::UpdateLaserRotation,
117 0.016f,
118 true // Loop
119 );
120
121 // Start the duration timer if not infinite
122 if (!bInfiniteDuration)
123 {
124 GetWorld()->GetTimerManager().SetTimer(
125 DurationTimerHandle,
126 this,
127 &UBaseLaserAttackAbility::FinishLaserAttack,
128 LaserDuration,
129 false
130 );
131 }
132 }
133 else
134 {
135 // If no lasers were spawned, end the ability immediately.
136 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, true);
137 }
138}

140void UBaseLaserAttackAbility::FinishLaserAttack()
141{
142 if (bIsFinished || !GetActorInfo().IsNetAuthority())
143 {
144 return;
145 }
146 bIsFinished = true;
147
148 // Clear all timers
149 GetWorld()->GetTimerManager().ClearTimer(DamageTimerHandle);
150 GetWorld()->GetTimerManager().ClearTimer(RotationTimerHandle);
151 GetWorld()->GetTimerManager().ClearTimer(DurationTimerHandle);
152
153 // Deactivate and destroy all spawned lasers
154 for (ALaserBeam* Laser : SpawnedLasers)
155 {
156 if (Laser)
157 {
158 Laser->DeactivateLaser();
159 Laser->Destroy();
160 }
161 }
162 SpawnedLasers.Empty();
163 InitialSpawnTransforms.Empty();
164
165 // End the ability itself
166 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, false);
167}

169TArray<FTransform> UBaseLaserAttackAbility::CalculateLaserSpawnTransforms()
170{
171 TArray<FTransform> SpawnTransforms;
172 AActor* AvatarActor = GetAvatarActorFromActorInfo();
173 if (!AvatarActor || NumberOfLasers <= 0) return SpawnTransforms;
174
175 const FVector SpawnCenter = AvatarActor->GetActorLocation();
176 const float AngleStep = 360.f / NumberOfLasers;
177
178 for (int32 i = 0; i < NumberOfLasers; ++i)
179 {
180 const float Angle = InitialRotationOffset + (i * AngleStep);
181 const FRotator Rotation(0.f, Angle, 0.f);
182
183 const FVector Location = SpawnCenter + Rotation.Vector() * SpawnRadius;
184
185 SpawnTransforms.Add(FTransform(Rotation, Location));
186 }
187
188 return SpawnTransforms;
189}

191void UBaseLaserAttackAbility::UpdateLaserRotation()
192{
193 if (!GetActorInfo().IsNetAuthority() || SpawnedLasers.Num() == 0) return;
194
195 AActor* AvatarActor = GetAvatarActorFromActorInfo();
196 if (!AvatarActor) return;
197
198 const float DeltaTime = GetWorld()->GetTimerManager().GetTimerRate(RotationTimerHandle);
199 CurrentRotationAngle = FMath::Fmod(CurrentRotationAngle + RotationSpeed * DeltaTime, 360.f);
200
201 const FVector CurrentAvatarLocation = AvatarActor->GetActorLocation();
202 const FRotator AdditiveRotation(0.f, CurrentRotationAngle, 0.f);
203
204 for (int32 i = 0; i < SpawnedLasers.Num(); ++i)
205 {
206 ALaserBeam* Laser = SpawnedLasers[i];
207 if (Laser && i < InitialSpawnTransforms.Num())
208 {
209 const FTransform& InitialTransform = InitialSpawnTransforms[i];
210
211 const FVector InitialOffset = InitialTransform.GetLocation() - AvatarInitialLocation;
212
213 const FVector RotatedOffset = AdditiveRotation.RotateVector(InitialOffset);
214
215 const FVector NewLocation = CurrentAvatarLocation + RotatedOffset;
216
217 const FRotator NewRotation = InitialTransform.GetRotation().Rotator() + AdditiveRotation;
218
219 FVector NewLaserEnd = NewLocation + NewRotation.Vector() * LaserLength;
220
221 Laser->SetLaserBeamEnd(NewLaserEnd);
222
223 Laser->SetActorLocationAndRotation(NewLocation, NewRotation);
224 }
225 }
226}
