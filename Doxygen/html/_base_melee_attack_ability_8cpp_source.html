23void UBaseMeleeAttackAbility::OnTraceEventReceived()
24{
25 // 서버 권한이 있는지 확인합니다. 피해 판정은 반드시 서버에서만 이루어져야 합니다.
26 if (!GetActorInfo().IsNetAuthority())
27 {
28 return;
29 }
30 // 나이아가라 이펙트 발생
31 const FVector SourceLocation = GetAvatarActorFromActorInfo()->GetActorLocation();
32 const FVector SourceDirection = GetAvatarActorFromActorInfo()->GetActorForwardVector();
33 FVector SpawnLocation = SourceLocation + SourceDirection * (AttackRange / 2.f);
34
35 UAbilitySystemComponent* SourceASC = GetAbilitySystemComponentFromActorInfo();
36 if (VFXCueTag.IsValid() && SourceASC)
37 {
38 FGameplayCueParameters CueParams;
39 CueParams.Location = SpawnLocation;
40 CueParams.Normal = SourceDirection.GetSafeNormal();
41 CueParams.NormalizedMagnitude = 0.f; // 공격 이펙트 요청
42 CueParams.OriginalTag = VFXCueTag;
43 SourceASC->ExecuteGameplayCue(VFXCueTag, CueParams);
44 }
45
46
47 ACharacter* OwnerCharacter = Cast<ACharacter>(GetAvatarActorFromActorInfo());
48 if (!OwnerCharacter) return;
49
50 FVector StartLocation = OwnerCharacter->GetActorLocation();
51 FVector ForwardVector = OwnerCharacter->GetActorForwardVector();
52
53 TArray<AActor*> OverlappedActors;
54 TArray<AActor*> ActorsToIgnore;
55 ActorsToIgnore.Add(OwnerCharacter);
56
57 // 1. 구체 오버랩으로 잠재적 타겟 수집
58 UKismetSystemLibrary::SphereOverlapActors(
59 GetWorld(),
60 StartLocation,
61 AttackRange,
62 TArray<TEnumAsByte<EObjectTypeQuery>>(),
63 AActor::StaticClass(),
64 ActorsToIgnore,
65 OverlappedActors
66 );
67
68 // 디버깅용 원뿔 그리기
69 if (bDrawDebugLine)
70 {
71 DrawDebugCone(
72 GetWorld(), StartLocation, ForwardVector, AttackRange,
73 FMath::DegreesToRadians(AttackAngle / 2.0f),
74 FMath::DegreesToRadians(AttackAngle / 2.0f),
75 12, FColor::Red, false, 2.0f, 0, 1.0f
76 );
77 }
78
79 TArray<AActor*> HitActors;
80 for (AActor* TargetActor : OverlappedActors)
81 {
82 if (!TargetActor || !UAbilitySystemBlueprintLibrary::GetAbilitySystemComponent(TargetActor)) continue;
83
84 // 2. 원뿔 범위 내에 있는지 내적(dot product)으로 확인
85 FVector DirectionToTarget = (TargetActor->GetActorLocation() - StartLocation).GetSafeNormal();
86 float DotProduct = FVector::DotProduct(ForwardVector, DirectionToTarget);
87 float AngleInDegrees = FMath::RadiansToDegrees(FMath::Acos(DotProduct));
88
89 if (AngleInDegrees <= AttackAngle / 2.0f)
90 {
91 HitActors.Add(TargetActor);
92 }
93 }
94
95 // 3. 유효 타겟에게 데미지 이펙트 적용
96 if (HitActors.Num() > 0 && PayloadEffectClass)
97 {
98 for (AActor* HitActor : HitActors)
99 {
100 IAbilitySystemInterface* TargetASI = Cast<IAbilitySystemInterface>(HitActor);
101 if (!TargetASI) return;
102
103 UAbilitySystemComponent* TargetASC = TargetASI->GetAbilitySystemComponent();
104 if (!TargetASI) return;
105
106 FGameplayEffectContextHandle EffectContext = SourceASC->MakeEffectContext();
107 EffectContext.AddSourceObject(OwnerCharacter);
108 FGameplayEffectSpecHandle SpecHandle = SourceASC->MakeOutgoingSpec(PayloadEffectClass, GetAbilityLevel(), EffectContext);
109
110 if (!SpecHandle.IsValid()) continue;
111
112 const FGameplayTag FinalSkillDamageTag = FGameplayTag::RequestGameplayTag(FName("Data.Damage.FinalSkillDamage"));
113
114 FGameplayEffectSpec* Spec = SpecHandle.Data.Get();
115 if (Spec)
116 {
117 Spec->SetSetByCallerMagnitude(FinalSkillDamageTag, Damage);
118 }
119
120 TargetASC->ApplyGameplayEffectSpecToSelf(*SpecHandle.Data.Get());
121 }
122 }
123}

130void UBaseMeleeAttackAbility::SpawnCircleIndicator()
131{
132 if (!GetActorInfo().IsNetAuthority())
133 {
134 return;
135 }
136 // 나이아가라 이펙트 발생
137 const FVector SourceLocation = GetAvatarActorFromActorInfo()->GetActorLocation();
138 const FVector SourceDirection = GetAvatarActorFromActorInfo()->GetActorForwardVector();
139 FVector SpawnLocation = SourceLocation/* + SourceDirection * (AttackRange / 2.f)*/;
140 FTransform SpawnTransform = FTransform(SpawnLocation);
141
142 ACircleIndicatorActor* Indicator = GetWorld()->SpawnActorDeferred<ACircleIndicatorActor>(
143 CircleIndicatorClass,
144 SpawnTransform,
145 GetAvatarActorFromActorInfo(),
146 Cast<APawn>(GetAvatarActorFromActorInfo()),
147 ESpawnActorCollisionHandlingMethod::AlwaysSpawn
148 );
149
150 if (Indicator)
151 {
152 UGameplayStatics::FinishSpawningActor(Indicator, SpawnTransform);
153 Indicator->InitializeIndicator(SpawnLocation, IndicatorTime, AttackRange);
154 }
155}
