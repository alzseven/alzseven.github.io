86void UKPTCheatManager::AddSkill(FName SkillID, int32 PlayerIndex)
87{
88 APlayerController* PlayerController = UGameplayStatics::GetPlayerController(GetWorld(), PlayerIndex);
89 if (!IsValid(PlayerController)) return;
90 APlayerCharacter* PlayerCharacter = Cast<APlayerCharacter>(PlayerController->GetCharacter());
91 if (!PlayerCharacter) return;
92 USkillManagerComponent* SkillManagerComponent = PlayerCharacter->FindComponentByClass<USkillManagerComponent>();
93 if (SkillManagerComponent)
94 {
95 SkillManagerComponent->AddSkill(SkillID);
96 UE_LOG(LogTemp, Log, TEXT("Exec command: AddSkill %s executed."), *SkillID.ToString());
97 }
98 else
99 {
100 UE_LOG(LogTemp, Warning, TEXT("SkillManagerComponent not found on %s."), *GetName());
101 }
102}
