38void AKPTPlayerState::NotifyComponentReady_Implementation(UActorComponent* Component)
39{
40 if (UInventoryManager* Inventory = Cast<UInventoryManager>(Component))
41 {
42 OnInventoryManagerReady.Broadcast(Inventory);
43 }
44 else if (UPlayerWalletComponent* Wallet = Cast<UPlayerWalletComponent>(Component))
45 {
46 OnWalletComponentReady.Broadcast(Wallet);
47 }
48 else if (UPlayerShopStateComponent* ShopState = Cast<UPlayerShopStateComponent>(Component))
49 {
50 OnShopStateComponentReady.Broadcast(ShopState);
51 }
52 else if (UPlayerRewardStateComponent* RewardState = Cast<UPlayerRewardStateComponent>(Component))
53 {
54 OnRewardStateComponentReady.Broadcast(RewardState);
55 }
56 else if (UKPTLobbyStateComponent* LobbyState = Cast<UKPTLobbyStateComponent>(Component))
57 {
58 OnLobbyStateComponentReady.Broadcast(LobbyState);
59 }
60}

172void AKPTPlayerState::OnDeadTagChanged(const FGameplayTag Tag, int32 NewCount)
173{
174 // NewCount가 1 이상이면 태그가 추가된 것(사망), 0이면 제거된 것(부활)입니다.
175 bool bIsNowDead = NewCount > 0;
176
177 UE_LOG(LogTemp, Warning, TEXT("Player %s Dead State Changed: %s"), *GetPlayerName(), bIsNowDead ? TEXT("Dead") : TEXT("Alive"));
178
179 // 여기에서 사망 관련 UI 위젯을 띄우거나 숨기는 로직을 실행합니다.
180 OnPlayerDiedChanged.Broadcast(bIsNowDead);
181}
