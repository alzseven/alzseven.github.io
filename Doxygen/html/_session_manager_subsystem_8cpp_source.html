42void USessionManagerSubsystem::HostSession(int32 NumPublicConnections, bool bIsLANMatch, const FString& MapPath)
43{
44 IOnlineSessionPtr SessionInt = GetSessionInterface();
45 if (!SessionInt.IsValid())
46 {
47 OnHostSessionComplete.Broadcast(false);
48 return;
49 }
50
51 // 세션이 이미 존재하면, 새로 생성하지 않고 실패를 알립니다.
52 if(SessionInt->GetNamedSession(NAME_GameSession) != nullptr)
53 {
54 UE_LOG(LogOnlineSession, Warning, TEXT("HostSession failed: A session already exists. Destroy it first."));
55 OnHostSessionComplete.Broadcast(false);
56 return;
57 }
58
59 PendingTravelURL = MapPath;
60
61 FOnlineSessionSettings SessionSettings;
62 SessionSettings.NumPublicConnections = NumPublicConnections;
63 SessionSettings.bShouldAdvertise = true;
64 SessionSettings.bAllowJoinInProgress = true;
65 SessionSettings.bIsLANMatch = bIsLANMatch;
66 SessionSettings.bUsesPresence = !bIsLANMatch; // LAN 매치가 아닐 때만 Presence 사용
67 SessionSettings.Set(GetGameInstance()->GetOnlinePlatformName(), FString("KPT Session"), EOnlineDataAdvertisementType::ViaOnlineServiceAndPing);
68
69 ULocalPlayer* LocalPlayer = GetGameInstance()->GetFirstGamePlayer();
70 if (!LocalPlayer)
71 {
72 OnHostSessionComplete.Broadcast(false);
73 return;
74 }
75
76 SessionInt->CreateSession(*LocalPlayer->GetPreferredUniqueNetId(), NAME_GameSession, SessionSettings);
77}

79void USessionManagerSubsystem::FindSessions(int32 MaxSearchResults, bool bIsLANQuery)
80{
81 IOnlineSessionPtr SessionInt = GetSessionInterface();
82 if (!SessionInt.IsValid())
83 {
84 OnFindSessionsCompleteResult.Broadcast(false, TArray<FOnlineSessionSearchResult>());
85 return;
86 }
87
88 SessionSearch = MakeShareable(new FOnlineSessionSearch());
89 SessionSearch->MaxSearchResults = MaxSearchResults;
90 SessionSearch->bIsLanQuery = bIsLANQuery;
91
92 // LAN 쿼리일 때는 불필요한 검색 조건을 모두 제거하여 신뢰도를 높입니다.
93 if (!bIsLANQuery)
94 {
95 SessionSearch->QuerySettings.Set(FName(TEXT("Public")), true, EOnlineComparisonOp::Equals);
96 }
97
98 ULocalPlayer* LocalPlayer = GetGameInstance()->GetFirstGamePlayer();
99 if (!LocalPlayer)
100 {
101 OnFindSessionsCompleteResult.Broadcast(false, TArray<FOnlineSessionSearchResult>());
102 return;
103 }
104 SessionInt->FindSessions(*LocalPlayer->GetPreferredUniqueNetId(), SessionSearch.ToSharedRef());
105}

151void USessionManagerSubsystem::OnJoinSessionComplete(FName SessionName, EOnJoinSessionCompleteResult::Type Result)
152{
153 OnJoinSessionCompleteResult.Broadcast(Result);
154 if (Result == EOnJoinSessionCompleteResult::Success)
155 {
156 IOnlineSessionPtr SessionInt = GetSessionInterface();
157 if (SessionInt.IsValid())
158 {
159 FString ConnectString;
160 if(SessionInt->GetResolvedConnectString(SessionName, ConnectString))
161 {
162 APlayerController* PC = GetGameInstance()->GetFirstLocalPlayerController();
163 if (PC)
164 {
165 PC->ClientTravel(ConnectString, ETravelType::TRAVEL_Absolute);
166 }
167 }
168 }
169 }
