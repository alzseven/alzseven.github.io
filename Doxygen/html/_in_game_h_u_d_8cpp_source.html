129void UInGameHUD::UpdateSkillSlotData(const TArray<FName>& SkillNames)
130{
131 if (SkillDataTable)
132 {
133 for (int i = 0; i < SkillNames.Num(); ++i)
134 {
135 if (i >= SkillSlots.Num())
136 {
137 break;
138 }
139
140 FSkillData* SkillData = SkillDataTable->FindRow<FSkillData>(SkillNames[i], "");
141
142 if (SkillData)
143 {
144 SkillSlots[i]->UpdateSlotVisual(*SkillData);
145 }
146 }
147 }

167void UInGameHUD::StateTagsChangedSetPhaseText(const FGameplayTagContainer& CurrentTags)
168{
169 // 블루프린트에서 하드코딩된 'InGame.Phase.Night' 태그를 가져옵니다.
170 const FGameplayTag NightTag = FGameplayTag::RequestGameplayTag(FName("InGame.Phase.Night"));
171
172 // 위젯과 태그가 유효한지 확인합니다.
173 if (!Timer || !NightTag.IsValid())
174 {
175 return;
176 }
177
178 // 블루프린트의 If(Branch) 노드: CurrentTags에 NightTag가 없는 경우에만 실행합니다.
179 if (!CurrentTags.HasTag(NightTag))
180 {
181 // 블루프린트의 ForEachLoop: 컨테이너의 모든 태그를 순회합니다.
182 // 참고: 이 로직은 루프가 돌 때마다 텍스트를 계속 덮어쓰므로,
183 // 최종적으로는 컨테이너의 마지막 태그 이름만 표시됩니다.
184 FText TextToSet;
185 for (const FGameplayTag& Tag : CurrentTags)
186 {
187 // 블루프린트의 GetTagName -> Conv_NameToText 노드에 해당합니다.
188 // 태그를 FText로 변환합니다.
189 TextToSet = FText::FromName(Tag.GetTagName());
190 }
191
192 // 블루프린트의 SetText 노드: 변환된 텍스트를 EditableTextBox에 설정합니다.
193 // 루프가 모두 돈 후 마지막으로 변환된 텍스트가 설정됩니다.
194 if (!TextToSet.IsEmpty())
195 {
196 Timer->SetText(TextToSet);
197 }
198 }
199}

220void UInGameHUD::BindToGameStateDelegates(AKPTInGameState* GameState)
221{
222 if (!GameState)
223 {
224 return;
225 }
226
227 if (UTeamStatsComponent* Stats = GameState->TeamStatsComponent)
228 {
229 TeamStatsComponentsReadyHandle(GameState->TeamStatsComponent);
230 }
231
232 else
233 {
234 GameState->OnTeamStatsComponentReady.RemoveDynamic(this, &UInGameHUD::TeamStatsComponentsReadyHandle);
235 GameState->OnTeamStatsComponentReady.AddDynamic(this, &UInGameHUD::TeamStatsComponentsReadyHandle);
236 }
237
238 if (UInGamePhaseManagerComponent* PhaseManager = GameState->InGamePhaseManager)
239 {
240 InGamePhaseManagerReadyHandle(GameState->InGamePhaseManager);
241 }
242
243 else
244 {
245 GameState->OnInGamePhaseManagerReady.RemoveDynamic(this, &UInGameHUD::InGamePhaseManagerReadyHandle);
246 GameState->OnInGamePhaseManagerReady.AddDynamic(this, &UInGameHUD::InGamePhaseManagerReadyHandle);
247 }
