27bool AGCN_SkillVFXManager::OnExecute_Implementation(AActor* MyTarget, const FGameplayCueParameters& Parameters)
28{
29 Super::OnExecute_Implementation(MyTarget, Parameters);
30 if (!HasAuthority())
31 {
32 int a = 0;
33 }
34 if (!VFXDataTable)
35 {
36 UE_LOG(LogTemp, Warning, TEXT("AGCN_SkillVFXManager: VFXDataTable이 할당되지 않았습니다!"));
37 return false;
38 }
39
40 const FName RowName = Parameters.OriginalTag.GetTagName();
41 const FSkillVFXDataRow* VFXDataRow = VFXDataTable->FindRow<FSkillVFXDataRow>(RowName, TEXT(""));
42
43 if (!VFXDataRow)
44 {
45 UE_LOG(LogTemp, Warning, TEXT("AGCN_SkillVFXManager: 데이터 테이블에서 %s 행을 찾을 수 없습니다."), *RowName.ToString());
46 return false;
47 }
48
49 TSoftObjectPtr<UNiagaraSystem> SelectedVFX;
50 TSoftObjectPtr<USoundBase> SelectedSFX;
51
52 if (FMath::IsNearlyZero(Parameters.NormalizedMagnitude))
53 {
54 SelectedVFX = VFXDataRow->ShootVFX;
55 SelectedSFX = VFXDataRow->ShootSFX;
56 }
57 else if (FMath::IsNearlyEqual(1.0, Parameters.NormalizedMagnitude)) //피격
58 {
59 SelectedVFX = VFXDataRow->HitVFX;
60 SelectedSFX = VFXDataRow->HitSFX;
61 }
62 else if (FMath::IsNearlyEqual(2.0, Parameters.NormalizedMagnitude))// 2.f == 소멸
63 {
64 SelectedVFX = VFXDataRow->HitVFX;
65 }
66
67 if (UNiagaraSystem** PtrNiagara = VFXLoadMap.Find(SelectedVFX))
68 {
69 if (*PtrNiagara)
70 {
71 //Multicast_PlayVFX(Parameters,*PtrNiagara);
72 UNiagaraFunctionLibrary::SpawnSystemAtLocation(
73 GetWorld(),
74 *PtrNiagara,
75 Parameters.Location,
76 Parameters.Normal.Rotation()
77 );
78 }
79 }
80
81 if (USoundBase** PtrSound = SFXLoadMap.Find(SelectedSFX))
82 {
83 if (*PtrSound)
84 {
85 //Multicast_PlaySFX(Parameters,*PtrSound);
86 UGameplayStatics::SpawnSoundAtLocation(
87 GetWorld(),
88 *PtrSound,
89 Parameters.Location // GameplayCueParameters에 있는 위치 정보를 사용
90 );
91 }
92 }
93 return true;
94}

96void AGCN_SkillVFXManager::LoadAsset()
97{
98 if (!VFXDataTable)
99 {
100 UE_LOG(LogTemp, Warning, TEXT("AGCN_SkillVFXManager: VFXDataTable이 할당되지 않았습니다!"));
101 return;
102 }
103 for (auto Row: VFXDataTable->GetRowMap())
104 {
105 FSkillVFXDataRow* RowData = reinterpret_cast<FSkillVFXDataRow*>(Row.Value);
106 TSoftObjectPtr<UNiagaraSystem> SelectedShootVFX = RowData->ShootVFX;
107 TSoftObjectPtr<UNiagaraSystem> SelectedHitVFX = RowData->HitVFX;
108 TSoftObjectPtr<USoundBase> SelectedShootSFX = RowData->ShootSFX;
109 TSoftObjectPtr<USoundBase> SelectedHitSFX = RowData->HitSFX;
110
111 if (!SelectedShootVFX.IsNull() && !VFXLoadMap.Contains(SelectedShootVFX))
112 {
113 UNiagaraSystem* Niagara = SelectedShootVFX.LoadSynchronous();
114 VFXLoadMap.Emplace(SelectedShootVFX,Niagara);
115 }
116
117 if (!SelectedHitVFX.IsNull() && !VFXLoadMap.Contains(SelectedHitVFX))
118 {
119 UNiagaraSystem* Niagara = SelectedHitVFX.LoadSynchronous();
120 VFXLoadMap.Emplace(SelectedHitVFX,Niagara);
121 }
122
123 if (!SelectedShootSFX.IsNull() && !SFXLoadMap.Contains(SelectedShootSFX))
124 {
125 USoundBase* SoundBase = SelectedShootSFX.LoadSynchronous();
126 SFXLoadMap.Emplace(SelectedShootSFX,SoundBase);
127 }
128
129 if (!SelectedHitSFX.IsNull() && !SFXLoadMap.Contains(SelectedHitSFX))
130 {
131 USoundBase* SoundBase = SelectedHitSFX.LoadSynchronous();
132 SFXLoadMap.Emplace(SelectedHitSFX,SoundBase);
133 }
134 }
135}
