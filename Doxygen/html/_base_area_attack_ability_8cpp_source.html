25TArray<FVector> UBaseAreaAttackAbility::CalculateTargetLocation()
26{
27 TArray<FVector> TargetLocations;
28
29 // 플레이어들 찾기
30 TArray<AActor*> Players;
31 UGameplayStatics::GetAllActorsWithTag(GetWorld(), FName("Player"), Players);
32
33 if (bTargetMultiplePlayers)
34 {
35 // 여러 플레이어 타겟
36 for (int32 i = 0; i < FMath::Min(NumberOfAreas, Players.Num()); i++)
37 {
38 if (Players.IsValidIndex(i))
39 {
40 TargetLocations.Add(Players[i]->GetActorLocation());
41 }
42 }
43 }
44 else
45 {
46 // 가장 가까운 플레이어 하나만 타겟
47 if (Players.Num() > 0)
48 {
49 AActor* ClosestPlayer = nullptr;
50 float ClosestDistance = MAX_FLT;
51
52 FVector BossLocation = GetAvatarActorFromActorInfo()->GetActorLocation();
53 for (AActor* Player : Players)
54 {
55 float Distance = FVector::Dist(BossLocation, Player->GetActorLocation());
56 if (Distance < ClosestDistance)
57 {
58 ClosestDistance = Distance;
59 ClosestPlayer = Player;
60 }
61 }
62
63 if (ClosestPlayer)
64 {
65 for (int32 i = 0; i < NumberOfAreas; i++)
66 {
67 // 플레이어 주변에 랜덤하게 배치
68 FVector RandomOffset = FVector(
69 FMath::RandRange(-AreaOffset, AreaOffset),
70 FMath::RandRange(-AreaOffset, AreaOffset),
71 0.0f
72 );
73 TargetLocations.Add(ClosestPlayer->GetActorLocation() + RandomOffset);
74 }
75 }
76 }
77 }
78
79 return TargetLocations;
80}

82void UBaseAreaAttackAbility::SpawnAreaHazards()
83{
84 if (!AreaHazardClass || !HasAuthority(&CurrentActivationInfo))
85 {
86 OnAbilityEnd();
87 return;
88 }
89
90 TArray<FVector> TargetLocations = CalculateTargetLocation();
91 for (const FVector& Location : TargetLocations)
92 {
93 FActorSpawnParameters SpawnParams;
94 SpawnParams.Owner = GetAvatarActorFromActorInfo();
95 SpawnParams.Instigator = Cast<APawn>(GetAvatarActorFromActorInfo());
96
97 // 바닥에 스폰
98 FVector SpawnLocation = Location;
99 SpawnLocation.Z = 0.0f; // 또는 라인트레이스로 바닥 찾기
100
101 if (AAreaHazard* Hazard = GetWorld()->SpawnActor<AAreaHazard>(
102 AreaHazardClass, SpawnLocation, FRotator::ZeroRotator, SpawnParams))
103 {
104 // 장판 설정
105 Hazard->SetHazardParams(
106 AreaRadius,
107 ActivationDelay,
108 AreaDuration,
109 PayloadEffectClass,
110 Damage,
111 DamageGameplayTag,
112 GetAbilitySystemComponentFromActorInfo(),
113 VFXTag
114 );
115
116 SpawnedHazards.Add(Hazard);
117 }
118 }
119
120 // 모든 장판이 사라질 때까지 대기
121 FTimerHandle EndTimerHandle;
122 GetWorld()->GetTimerManager().SetTimer(EndTimerHandle, this,
123 &UBaseAreaAttackAbility::OnAbilityEnd, ActivationDelay + AreaDuration + 0.5f);
124}

126void UBaseAreaAttackAbility::OnAbilityEnd()
127{
128 for (AAreaHazard* Hazard : SpawnedHazards)
129 {
130 if (IsValid(Hazard))
131 {
132 Hazard->Destroy();
133 }
134 }
135 SpawnedHazards.Empty();
136
137 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, false);
138}
