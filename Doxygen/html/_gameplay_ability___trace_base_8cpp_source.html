69void UGameplayAbility_TraceBase::ApplyEffectToTarget(AActor* Target, float DamageAmount)
70{
71 if (!PayloadEffectClass || !Target) return;
72
73 UAbilitySystemComponent* SourceASC = GetAbilitySystemComponentFromActorInfo();
74 UAbilitySystemComponent* TargetASC = UAbilitySystemBlueprintLibrary::GetAbilitySystemComponent(Target);
75 if (!SourceASC || !TargetASC) return;
76
77 FGameplayEffectContextHandle EffectContext = SourceASC->MakeEffectContext();
78 EffectContext.AddSourceObject(SourceASC->GetOwner());
79 FGameplayEffectSpecHandle SpecHandle = SourceASC->MakeOutgoingSpec(PayloadEffectClass, GetAbilityLevel(), EffectContext);
80 if (SpecHandle.IsValid())
81 {
82 UAbilitySystemBlueprintLibrary::AssignTagSetByCallerMagnitude(SpecHandle, DamageTag, BaseDamage);
