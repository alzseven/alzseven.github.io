9void UStageManagerSubsystem::StartStage(int32 InStageIndex)
10{
11 if (!StageDataTable)
12 {
13 UE_LOG(LogTemp, Error, TEXT("StageDataTable is null!"));
14 return;
15 }
16
17 // Row를 찾는다. (FindRow의 2번째 인자는 ContextString, 디버깅용)
18 static const FString ContextString(TEXT("UStageManagerSubsystem::StartStage"));
19
20 TArray<FStageDataRow*> Rows;
21 StageDataTable->GetAllRows<FStageDataRow>(ContextString, Rows);
22
23 const int32 DataTableIndex = InStageIndex - 1;
24
25 check(Rows.IsEmpty() == false);
26
27 if (Rows.IsValidIndex(DataTableIndex) == false)
28 {
29 UE_LOG(LogTemp, Error, TEXT("다음 스테이지 정보가 없습니다."));
30 return;
31 }
32
33 CurrentStageData = Rows[DataTableIndex];
34
35 check(CurrentStageData);
36
37 //Spawner를 찾아서 데이터 셋팅하기.
38
39 SetStageDataToSpawner();
40
41 //Phase가 있다면 등록
42 if (CurrentStageData->StagePhases.IsEmpty() == false)
43 {
44 for (int32 i = 0; i < CurrentStageData->StagePhases.Num(); ++i)
45 {
46 FTimerHandle Handle;
47 GetWorld()->GetTimerManager().SetTimer(
48 Handle, this, &UStageManagerSubsystem::StartPhase, CurrentStageData->StagePhases[i].ActionStartTime, false
49 );
50 PhaseTimerHandles.Add(Handle);
51 }
52 }
53}

69void UStageManagerSubsystem::StartPhase()
70{
71 check(CurrentStageData != nullptr);

74 return;
75
76 const FPhaseActionData& PhaseData = CurrentStageData->StagePhases[CurrentPhaseIndex];

90void UStageManagerSubsystem::HandleBossPhase(const FBossPhaseData& BossPhaseData)
91{
92 if (IsValid(BossPhaseData.BossClass) == false)
93 return;
94
95 //TODO 보스생성
96
97 TArray<AActor*> FindActor;
98 UGameplayStatics::GetAllActorsOfClass(GetWorld(), AMonsterSpawnActor::StaticClass(), FindActor);
99
100 if (FindActor.IsEmpty())
101 {
102 UE_LOG(LogTemp, Error, TEXT("보스 몬스터 스폰포인트가 없습니다."));
103
104 //어떤 보스 클래스를 어디에 생성할지?
105 GetWorld()->SpawnActor<AActor>(BossPhaseData.BossClass, FVector(540.000000,660.000000,500.785363), FRotator(0.f, 180.f, 0.));
106 }
107 else
108 {
109 const FVector& SpawnLocation = FindActor[0]->GetActorLocation();
110 const FRotator& SpawnRotator = FindActor[0]->GetActorRotation();
111
112 //어떤 보스 클래스를 어디에 생성할지?
113 GetWorld()->SpawnActor<AActor>(BossPhaseData.BossClass, SpawnLocation, SpawnRotator);
114 }
115 UE_LOG(LogTemp, Log, TEXT("Boss Phase Execute"));
116}
