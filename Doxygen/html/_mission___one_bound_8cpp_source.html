69void AMission_OneBound::OnBallEnteredIndicator(ABouncingBall* Ball)
70{
71
72 if (!HasAuthority() || !Ball) return;
73
74 // 공이 들어왔으므로, 일단 움직임을 멈추게 합니다.

99void AMission_OneBound::PlanNextBounce()
100{
101 if (!HasAuthority() || !BouncingBall || !LandingIndicator) return;
102 if (CurrentLandingIndicator)
103 {
104 CurrentLandingIndicator->Destroy();
105 }
106
107 FVector NextLandingPoint = GetActorLocation();
108 UNavigationSystemV1* NavSys = FNavigationSystem::GetCurrent<UNavigationSystemV1>(GetWorld());
109 if (NavSys)
110 {
111 FNavLocation RandomPoint;
112 if (NavSys->GetRandomReachablePointInRadius(CurrentBall->GetActorLocation(), MissionRadius, RandomPoint))
113 {
114 NextLandingPoint = RandomPoint.Location;
115 }
116 }
117
118 CurrentLandingIndicator = GetWorld()->SpawnActor<ALandingIndicator>(LandingIndicator, NextLandingPoint, FRotator::ZeroRotator);
119 if(CurrentLandingIndicator)
120 {
121 CurrentLandingIndicator->SetMissionOwner(this);
122 }
123 CurrentBall->LaunchTowards(NextLandingPoint);
124}
