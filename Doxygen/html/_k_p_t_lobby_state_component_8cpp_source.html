25void UKPTLobbyStateComponent::SetReadyStatus(bool bInIsReady)
26{
27 if (GetOwner()->HasAuthority())
28 {
29 bIsReady = bInIsReady;
30 BroadcastOnReadyStatusChanged(); // 서버에서도 호출
31
32 // GameMode에게 상태 변경 알림
33 if (AKPTPlayerState* PS = Cast<AKPTPlayerState>(GetOwner()))
34 {
35 if (AKPTLobbyGameMode* GM = GetWorld()->GetAuthGameMode<AKPTLobbyGameMode>())
36 {
37 GM->UpdatePlayerReadyState(PS->GetPlayerController(), bIsReady);
38 }
39 }
40 }
