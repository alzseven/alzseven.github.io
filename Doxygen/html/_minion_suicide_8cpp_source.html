112void UMinionSuicide::ApplyExplosionDamage()
113{
114 AActor* AvatarActor = GetAvatarActorFromActorInfo();
115 if (AvatarActor == nullptr)
116 return;
117
118 TArray<FHitResult> OutHits;
119 TArray<TEnumAsByte<EObjectTypeQuery>> ObjectTypes;
120
121 // Pawn 및 Enemy 채널을 대상으로 함
122 ObjectTypes.Add(UEngineTypes::ConvertToObjectType(ECC_Pawn));
123 //ObjectTypes.Add(UEngineTypes::ConvertToObjectType(ECC_GameTraceChannel3)); // Enemy
124
125 TArray<AActor*> IgnoreActors;
126 IgnoreActors.Add(AvatarActor); // 혹은 this->GetOuterAActor() 등
127
128 FVector Start = AvatarActor->GetActorLocation();
129
130 bool bHit = UKismetSystemLibrary::SphereTraceMultiForObjects(
131 this,
132 Start,
133 Start,
134 ExplosionRadius,
135 ObjectTypes,
136 false, // TraceComplex
137 IgnoreActors,
138 EDrawDebugTrace::None, // 디버그 시각화
139 OutHits,
140 true // IgnoreSelf
141 );
142
143 if (!bHit)
144 return;
145
146 TSet<AActor*> UniqueActors;
147
148 for (const FHitResult& Hit : OutHits)
149 {
150 AActor* HitActor = Hit.GetActor();
151 if (HitActor == nullptr)
152 continue;
153
154 if (UniqueActors.Contains(HitActor))
155 continue;
156
157 UniqueActors.Add(HitActor);
158 UE_LOG(LogTemp, Log, TEXT("Unique hit: %s"), *HitActor->GetName());
159
160 // 데미지 처리 등
161 UAbilitySystemComponent* OwnerASC = UAbilitySystemGlobals::GetAbilitySystemComponentFromActor(AvatarActor);
162 UAbilitySystemComponent* TargetASC = UAbilitySystemGlobals::GetAbilitySystemComponentFromActor(HitActor);
163
164 if (!TargetASC || !OwnerASC)
165 continue;
166
167 float AttackPower = OwnerASC->GetNumericAttribute(UKPTBaseAttributeSet::GetAttackPowerAttribute());
168
169 FGameplayEffectSpecHandle SpecHandle = OwnerASC->MakeOutgoingSpec(DamageEffectClass, 1.0f, OwnerASC->MakeEffectContext());
170
171 if (SpecHandle.IsValid())
172 {
173 // SetByCaller 방식으로 공격력 전달
174 SpecHandle.Data->SetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("Data.Damage.FinalSkillDamage")), 5000.f);
175
176 SpecHandle.Data->SetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("Data.Damage.IsCritical")), 0);
177
178 SpecHandle.Data->SetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("Data.Damage.CriticalMultiplier")), 1.f);
179 }
180
181 if (TargetASC && SpecHandle.IsValid())
182 {
183 OwnerASC->ApplyGameplayEffectSpecToTarget(*SpecHandle.Data.Get(), TargetASC);
184 }
185 }
186}
