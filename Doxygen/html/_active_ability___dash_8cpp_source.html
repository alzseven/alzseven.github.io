15void UActiveAbility_Dash::ActivateAbility(const FGameplayAbilitySpecHandle Handle,
16 const FGameplayAbilityActorInfo* ActorInfo, const FGameplayAbilityActivationInfo ActivationInfo,
17 const FGameplayEventData* TriggerEventData)
18{
19 Super::ActivateAbility(Handle, ActorInfo, ActivationInfo, TriggerEventData);
20
21 if (!CommitAbility(Handle, ActorInfo, ActivationInfo))
22 {
23 EndAbility(Handle, ActorInfo, ActivationInfo, true, true);
24 return;
25 }
26
27 // GameplayEvent로 전달된 데이터가 있는지 확인
28 if (TriggerEventData && TriggerEventData->TargetData.IsValid(0))
29 {
30 const FGameplayAbilityTargetData_LocationInfo* LocationData = static_cast<const FGameplayAbilityTargetData_LocationInfo*>(TriggerEventData->TargetData.Get(0));
31 if (LocationData)
32 {
33 // 데이터에서 방향 벡터를 꺼낸다
34 const FVector ActorLocation = LocationData->TargetLocation.LiteralTransform.GetLocation();
35 const FVector DashDirection = LocationData->TargetLocation.LiteralTransform.GetScale3D();
36
37 // 몽타주 재생 및 모션 워핑 타겟 설정
38 UAbilityTask_PlayMontageAndWait* MontageTask = UAbilityTask_PlayMontageAndWait::CreatePlayMontageAndWaitProxy(this, NAME_None, DashMontage);
39 MontageTask->OnCompleted.AddDynamic(this, &ThisClass::K2_EndAbility);
40 MontageTask->OnInterrupted.AddDynamic(this, &ThisClass::K2_EndAbility);
41 MontageTask->OnCancelled.AddDynamic(this, &ThisClass::K2_EndAbility);
42 MontageTask->OnBlendOut.AddDynamic(this, &ThisClass::K2_EndAbility);
43
44 // 타겟 설정 후 몽타주 활성화
45 SetMotionWarpingTarget(DashDirection, ActorLocation, BaseDashDistance);
46 MontageTask->ReadyForActivation();
47
48 return;
49 }
50 }
51
52 // 정상적으로 데이터가 오지 않으면 어빌리티를 취소
53 CancelAbility(Handle, ActorInfo, ActivationInfo, true);
54}

56void UActiveAbility_Dash::SetMotionWarpingTarget(const FVector& Direction, const FVector& StartLocation, float Distance)
57{
58 // HasAuthority()로 서버인지 클라이언트인지 명확히 구분하여 로그를 찍는다.
59 const FString RoleString = GetAvatarActorFromActorInfo()->HasAuthority() ? TEXT("Server") : TEXT("Client");
60
61 //UE_LOG(LogTemp, Warning, TEXT("[%s] SetMotionWarpingTarget called."), *RoleString);
62
63 APlayerCharacter* PlayerCharacter = Cast<APlayerCharacter>(GetAvatarActorFromActorInfo());
64 if (UMotionWarpingComponent* MWComp = PlayerCharacter->FindComponentByClass<UMotionWarpingComponent>())
65 {
66 //UE_LOG(LogTemp, Warning, TEXT("[%s] MotionWarpingComponent Found! Setting target."), *RoleString);
67
68 FVector TargetLocation = StartLocation + (Direction * Distance);
69 FRotator TargetRotation = Direction.Rotation();
70
71 //UE_LOG(LogTemp, Warning, TEXT("[%s] TargetRotation: %s"), *RoleString, *TargetRotation.ToString());
72 MWComp->AddOrUpdateWarpTargetFromLocationAndRotation(
73 FName("DashTarget"), TargetLocation, TargetRotation);
74 }
75 else
76 {
77 // 만약 컴포넌트를 못 찾았다면 이 로그가 출력될 것이다.
78 UE_LOG(LogTemp, Error, TEXT("[%s] MotionWarpingComponent NOT FOUND!"), *RoleString);
79 }
80}

82float UActiveAbility_Dash::CalcRootMotionScale()
83{
84 const int32 AbilityLevel = GetAbilityLevel();
85 float CurrentDashDistance = BaseDashDistance;
86 FRealCurve* FoundCurve = DashDistanceTable->FindCurve(TEXT("DashDistance"), TEXT("DashDistance Not Found"), true);
87 if (!FoundCurve)
88 {
89 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, true);
90 }
91 CurrentDashDistance = FoundCurve->Eval(AbilityLevel);
92
93 const float RootMotionScale = (BaseDashDistance > 0.f) ? (CurrentDashDistance / BaseDashDistance) : 1.f;
94 return RootMotionScale;
95}
