41void AKPTLobbyGameMode::PostLogin(APlayerController* NewPlayer)
42{
43 Super::PostLogin(NewPlayer);
44
45 // GameState와 PlayerState가 준비될 때까지 약간 기다립니다.
46 FTimerHandle TimerHandle;
47 FTimerDelegate TimerDelegate = FTimerDelegate::CreateLambda([this, NewPlayer]()
48 {
49 if (NewPlayer)
50 {
51 FString PlayerName = NewPlayer->PlayerState->GetPlayerName();
52
53 IOnlineSubsystem* Subsystem = Online::GetSubsystem(GetWorld());
54 if (Subsystem)
55 {
56 IOnlineIdentityPtr Identity = Subsystem->GetIdentityInterface();
57 if (Identity.IsValid())
58 {
59 // 새로 접속한 플레이어의 고유 ID를 가져옵니다.
60 TSharedPtr<const FUniqueNetId> UniqueId = NewPlayer->PlayerState->GetUniqueId().GetUniqueNetId();
61 if (UniqueId.IsValid())
62 {
63 // 고유 ID를 사용해 닉네임을 가져옵니다.
64 FString Nickname = Identity->GetPlayerNickname(*UniqueId);
65
66 // 가져온 닉네임으로 PlayerState의 이름을 설정합니다. 이 변경사항은 모든 클라이언트로 복제됩니다.
67 NewPlayer->PlayerState->SetPlayerName(Nickname.IsEmpty() ? FString("Player") : Nickname);
68 }
69 }
70 }
71
72 AKPTLobbyGameState* GS = GetGameState<AKPTLobbyGameState>();
73 AKPTPlayerState* PS = NewPlayer->GetPlayerState<AKPTPlayerState>();
74 if (GS && PS)
75 {
76 // 새 게임을 위해 PlayerState를 초기 상태로 되돌립니다.
77 PS->ResetForNewGame();
78
79 GS->AddPlayerToSlot(PS);
80
81 if (PS && CharacterDataTable)
82 {
83 // --- 여기가 핵심 변경점 ---
84 // SaveGame 로직 대신, 데이터 테이블의 모든 캐릭터 ID를 가져옵니다.
85 TArray<FGameplayTag> AllCharacterIDs;
86 TArray<FName> RowNames = CharacterDataTable->GetRowNames();
87 for (const FName& RowName : RowNames)
88 {
89 FCharacterData* RowData = CharacterDataTable->FindRow<FCharacterData>(RowName, TEXT(""));
90 if (RowData)
91 {
92 AllCharacterIDs.Add(RowData->CharacterID);
93 }
94 }
95
96 // 가져온 모든 캐릭터 ID 목록으로 PlayerState를 초기화합니다.
97 // PS->GetLobbyStateComponent()->InitializeUnlockedCharacters(AllCharacterIDs);
98 }
99 }
100 }
101 CheckLobbyState();
102 });
103
104 GetWorld()->GetTimerManager().SetTimer(TimerHandle, TimerDelegate, 0.2f, false);
105}

123void AKPTLobbyGameMode::UpdatePlayerReadyState(AController* PlayerController, bool bIsReady)
124{
125 AKPTLobbyGameState* GS = GetGameState<AKPTLobbyGameState>();
126 APlayerState* PS = PlayerController ? PlayerController->PlayerState : nullptr;
127
128 if (GS && PS)
129 {
130 // PlayerSlots 배열에서 해당 플레이어를 찾습니다.
131 for (FPlayerSlotInfo& Slot : GS->PlayerSlots)
132 {
133 if (Slot.PlayerState == PS)
134 {
135 // [핵심 수정] 슬롯 정보에 준비 상태를 직접 업데이트합니다.
136 Slot.bIsReady = bIsReady;
137 break;
138 }
139 }
140
141 // 배열 내용이 변경되었으므로, 모든 클라이언트에 변경 사항을 전파하고 델리게이트를 호출합니다.
142 GS->NotifyClientsWithSlotsUpdated();
143 GS->BroadcastSlotsUpdated();
144 }
145
146 // 기존 로비 상태 체크 로직은 그대로 둡니다.
147 CheckLobbyState();
148}

241void AKPTLobbyGameMode::CheckLobbyState()
242{
243 if (!CachedLobbyStateManager) return;
244
245 AKPTLobbyGameState* GS = GetGameState<AKPTLobbyGameState>();
246 if(!GS) return;
247
248 int32 TotalPlayers = GetNumPlayers();
249 int32 ReadyPlayers = 0;
250
251 // GameState의 PlayerSlots 배열을 순회하며 준비된 플레이어 수를 셉니다.
252 for (const FPlayerSlotInfo& Slot : GS->PlayerSlots)
253 {
254 if(Slot.bIsOccupied && Slot.bIsReady)
255 {
256 ReadyPlayers++;
257 }
258 }
259
260 CachedLobbyStateManager->SetReadyPlayerCount(ReadyPlayers);
261
262 if (TotalPlayers <= 0)
263 {
264 StopGameCountdown();
265 CachedLobbyStateManager->SetStateTags(FGameplayTagContainer(FGameplayTag::RequestGameplayTag(TEXT("Lobby.State.WaitingForPlayers"))));
266 return;
267 }
268
269 if (ReadyPlayers > 0 && ReadyPlayers == TotalPlayers)
270 {
271 CachedLobbyStateManager->SetStateTags(FGameplayTagContainer(FGameplayTag::RequestGameplayTag(TEXT("Lobby.State.AllPlayersReady"))));
272 StartGameCountdown(ShortCountdownDuration);
273 }
274 else if (ReadyPlayers > 0)
275 {
276 if (CachedLobbyStateManager->GetCurrentStateTags().HasTag(FGameplayTag::RequestGameplayTag(TEXT("Lobby.State.Countdown"))) == false)
277 {
278 CachedLobbyStateManager->SetStateTags(FGameplayTagContainer(FGameplayTag::RequestGameplayTag(TEXT("Lobby.State.Countdown"))));
279 StartGameCountdown(LongCountdownDuration);
280 }
281 }
282 else
283 {
284 CachedLobbyStateManager->SetStateTags(FGameplayTagContainer(FGameplayTag::RequestGameplayTag(TEXT("Lobby.State.WaitingForPlayers"))));
285 StopGameCountdown();
286 }
287 // 배열 내용이 변경되었으므로, 모든 클라이언트에 변경 사항을 전파하고 델리게이트를 호출합니다.
288 GS->NotifyClientsWithSlotsUpdated();
289 GS->BroadcastSlotsUpdated();
290}

292void AKPTLobbyGameMode::StartGameCountdown(float CountdownDuration)
293{
294 if (GetWorldTimerManager().IsTimerActive(GameStartTimer) && GetWorldTimerManager().GetTimerRemaining(GameStartTimer) == CountdownDuration)
295 {
296 return;
297 }
298
299 GetWorldTimerManager().SetTimer(GameStartTimer, this, &AKPTLobbyGameMode::StartGame, CountdownDuration, false);
300
301 if (CachedLobbyStateManager)
302 {
303 CachedLobbyStateManager->SetRemainingTime(CountdownDuration + GetWorld()->GetTimeSeconds());
304 }
305}

307void AKPTLobbyGameMode::StopGameCountdown()
308{
309 GetWorldTimerManager().ClearTimer(GameStartTimer);
310 if (CachedLobbyStateManager)
311 {
312 CachedLobbyStateManager->SetRemainingTime(0);
313 }
314}
