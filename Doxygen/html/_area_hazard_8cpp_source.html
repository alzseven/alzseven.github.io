78void AAreaHazard::ActivateHazard()
79{
80 if (!HasAuthority()) return;
81
82 bIsActive = true;
83 CollisionSphere->SetCollisionEnabled(ECollisionEnabled::QueryOnly);
84
85 TArray<AActor*> OverlappingActors;
86 CollisionSphere->GetOverlappingActors(OverlappingActors);
87
88 for (AActor* Actor : OverlappingActors)
89 {
90 if (Actor && Actor != SourceASC->GetAvatarActor())
91 {
92 ApplyEffectToTarget(Actor);
93 }
94 }
95
96 GetWorld()->GetTimerManager().SetTimer(
97 DeactivationTimerHandle,
98 this,
99 &AAreaHazard::DeactivateHazard,
100 Duration,
101 false
102 );
103
104 FGameplayCueParameters CueParams;
105 CueParams.Location = GetActorLocation();
106 CueParams.NormalizedMagnitude = 0.f;
107 CueParams.OriginalTag = VFXTag;
108 SourceASC->ExecuteGameplayCue(VFXTag, CueParams);
109}

130void AAreaHazard::ApplyEffectToTarget(AActor* TargetActor)
131{
132 if (!SourceASC || !DamageEffectClass || !TargetActor) return;
133
134 UAbilitySystemComponent* TargetASC = UAbilitySystemBlueprintLibrary::GetAbilitySystemComponent(TargetActor);
135 if (!TargetASC) return;
136
137 FGameplayEffectContextHandle EffectContext = SourceASC->MakeEffectContext();
138 EffectContext.AddSourceObject(SourceASC->GetOwner());
139 FGameplayEffectSpecHandle SpecHandle = SourceASC->MakeOutgoingSpec(DamageEffectClass, 1.f, EffectContext);
140 if (SpecHandle.IsValid())
141 {
142 UAbilitySystemBlueprintLibrary::AssignTagSetByCallerMagnitude(SpecHandle, DamageTag, Damage);
143
144 TargetASC->ApplyGameplayEffectSpecToSelf(*SpecHandle.Data.Get());
145 }
146}
