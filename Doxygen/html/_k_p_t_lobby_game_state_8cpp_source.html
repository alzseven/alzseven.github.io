15void AKPTLobbyGameState::BeginPlay()
16{
17 Super::BeginPlay();
18
19 FGameplayTag CenterTag = FGameplayTag::RequestGameplayTag(TEXT("Lobby.PlayerStart.Center"));
20 FGameplayTag LeftTag = FGameplayTag::RequestGameplayTag(TEXT("Lobby.PlayerStart.Left"));
21 FGameplayTag RightTag = FGameplayTag::RequestGameplayTag(TEXT("Lobby.PlayerStart.Right"));

24 CenterTag,
25 LeftTag,
26 RightTag

28

35 }
36}

44void AKPTLobbyGameState::AddPlayerToSlot(APlayerState* NewPlayer)
45{
46 if (!HasAuthority() || !NewPlayer) return;
47
48 // 비어있는 슬롯을 찾아서 플레이어를 추가합니다.
49 for (FPlayerSlotInfo& Slot : PlayerSlots)
50 {
51 if (!Slot.bIsOccupied)
52 {
53 Slot.PlayerState = NewPlayer;
54 Slot.bIsOccupied = true;
55
56 // 배열이 변경되었으므로, 클라이언트에 알리기 위해 RepNotify를 수동으로 호출합니다.
57 BroadcastSlotsUpdated();
58 MARK_PROPERTY_DIRTY_FROM_NAME(AKPTLobbyGameState, PlayerSlots, this);
59 return;
60 }
61 }
62}

64void AKPTLobbyGameState::RemovePlayerFromSlot(APlayerState* PlayerToRemove)
65{
66 if (!HasAuthority() || !PlayerToRemove) return;
67
68 // 나간 플레이어가 사용하던 슬롯을 찾아서 비웁니다.
69 for (FPlayerSlotInfo& Slot : PlayerSlots)
70 {
71 if (Slot.PlayerState == PlayerToRemove)
72 {
73 Slot.PlayerState = nullptr;
74 Slot.bIsOccupied = false;
75
76 BroadcastSlotsUpdated();
77 MARK_PROPERTY_DIRTY_FROM_NAME(AKPTLobbyGameState, PlayerSlots, this);
78 return;
79 }
80 }
81}
