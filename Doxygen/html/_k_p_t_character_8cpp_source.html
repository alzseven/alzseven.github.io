19AKPTCharacter::AKPTCharacter()
20{
21 // Set size for collision capsule
22 GetCapsuleComponent()->InitCapsuleSize(42.f, 96.0f);
23
24 // Don't rotate when the controller rotates. Let that just affect the camera.
25 bUseControllerRotationPitch = false;
26 bUseControllerRotationYaw = false;
27 bUseControllerRotationRoll = false;
28
29 // Configure character movement
30 GetCharacterMovement()->bOrientRotationToMovement = true; // Character moves in the direction of input...
31 GetCharacterMovement()->RotationRate = FRotator(0.0f, 500.0f, 0.0f); // ...at this rotation rate
32
33 // Note: For faster iteration times these variables, and many more, can be tweaked in the Character Blueprint
34 // instead of recompiling to adjust them
35 GetCharacterMovement()->JumpZVelocity = 700.f;
36 GetCharacterMovement()->AirControl = 0.35f;
37 GetCharacterMovement()->MaxWalkSpeed = 500.f;
38 GetCharacterMovement()->MinAnalogWalkSpeed = 20.f;
39 GetCharacterMovement()->BrakingDecelerationWalking = 2000.f;
40 GetCharacterMovement()->BrakingDecelerationFalling = 1500.0f;
41
42 // Create a camera boom (pulls in towards the player if there is a collision)
43 CameraBoom = CreateDefaultSubobject<USpringArmComponent>(TEXT("CameraBoom"));
44 CameraBoom->SetupAttachment(RootComponent);
45 CameraBoom->TargetArmLength = 400.0f; // The camera follows at this distance behind the character
46 CameraBoom->bUsePawnControlRotation = true; // Rotate the arm based on the controller
47
48 // Create a follow camera
49 FollowCamera = CreateDefaultSubobject<UCameraComponent>(TEXT("FollowCamera"));
50 FollowCamera->SetupAttachment(CameraBoom, USpringArmComponent::SocketName); // Attach the camera to the end of the boom and let the boom adjust to match the controller orientation
51 FollowCamera->bUsePawnControlRotation = false; // Camera does not rotate relative to arm
52
53 // Note: The skeletal mesh and anim blueprint references on the Mesh component (inherited from Character)
54 // are set in the derived blueprint asset named ThirdPersonCharacter (to avoid direct content references in C++)
55}

66void AKPTCharacter::SetupPlayerInputComponent(UInputComponent* PlayerInputComponent)
67{
68 // Add Input Mapping Context
69 if (APlayerController* PlayerController = Cast<APlayerController>(GetController()))
70 {
71 if (UEnhancedInputLocalPlayerSubsystem* Subsystem = ULocalPlayer::GetSubsystem<UEnhancedInputLocalPlayerSubsystem>(PlayerController->GetLocalPlayer()))
72 {
73 Subsystem->AddMappingContext(DefaultMappingContext, 0);
74 }
75 }
76
77 // Set up action bindings
78 if (UEnhancedInputComponent* EnhancedInputComponent = Cast<UEnhancedInputComponent>(PlayerInputComponent)) {
79
80 // Jumping
81 EnhancedInputComponent->BindAction(JumpAction, ETriggerEvent::Started, this, &ACharacter::Jump);
82 EnhancedInputComponent->BindAction(JumpAction, ETriggerEvent::Completed, this, &ACharacter::StopJumping);
83
84 // Moving
85 EnhancedInputComponent->BindAction(MoveAction, ETriggerEvent::Triggered, this, &AKPTCharacter::Move);
86
87 // Looking
88 EnhancedInputComponent->BindAction(LookAction, ETriggerEvent::Triggered, this, &AKPTCharacter::Look);
89 }
90 else
91 {
92 UE_LOG(LogTemplateCharacter, Error, TEXT("'%s' Failed to find an Enhanced Input component! This template is built to use the Enhanced Input system. If you intend to use the legacy system, then you will need to update this C++ file."), *GetNameSafe(this));
93 }
94}

96void AKPTCharacter::Move(const FInputActionValue& Value)
97{
98 // input is a Vector2D
99 FVector2D MovementVector = Value.Get<FVector2D>();
100
101 if (Controller != nullptr)
102 {
103 // find out which way is forward
104 const FRotator Rotation = Controller->GetControlRotation();
105 const FRotator YawRotation(0, Rotation.Yaw, 0);
106
107 // get forward vector
108 const FVector ForwardDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
109
110 // get right vector
111 const FVector RightDirection = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
112
113 // add movement
114 AddMovementInput(ForwardDirection, MovementVector.Y);
115 AddMovementInput(RightDirection, MovementVector.X);
116 }
117}

119void AKPTCharacter::Look(const FInputActionValue& Value)
120{
121 // input is a Vector2D
122 FVector2D LookAxisVector = Value.Get<FVector2D>();
123
124 if (Controller != nullptr)
125 {
126 // add yaw and pitch input to controller
127 AddControllerYawInput(LookAxisVector.X);
128 AddControllerPitchInput(LookAxisVector.Y);
129 }
130}
