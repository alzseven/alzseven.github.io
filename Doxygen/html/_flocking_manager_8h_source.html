33 int32 GetCellIndex(const FVector2D& Position) const
34 {
35 int32 X = FMath::FloorToInt(Position.X / CellSize);
36 int32 Y = FMath::FloorToInt(Position.Y / CellSize);
37
38 // 간단한 해싱
39 //return (X * 73856093) ^ (Y * 19349663) ^ (Z * 83492791);
40 return (X * 73856093) ^ (Y * 19349663);
41 }
42
43 TArray<int32> GetNearbyEntities(const FVector2D& Position, float Radius) const
44 {
45 TArray<int32> Result;
46 int32 CellRadius = FMath::CeilToInt(Radius / CellSize);
47
48 for (int32 dx = -CellRadius; dx <= CellRadius; dx++)
49 {
50 for (int32 dy = -CellRadius; dy <= CellRadius; dy++)
51 {
52 FVector2D CheckPos = FVector2D(Position.X, Position.Y) + FVector2D(dx, dy) * CellSize;
53 int32 CellIndex = GetCellIndex(CheckPos);
54
55 if (const TArray<int32>* Entities = CellToEntities.Find(CellIndex))
56 {
57 Result.Append(*Entities);
58 }
59 }
60 }
61
62 return Result;
63 }
