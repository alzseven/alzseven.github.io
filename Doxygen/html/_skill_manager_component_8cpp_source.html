22bool USkillManagerComponent::AddSkill(FName SkillID)
23{
24 if (!GetOwner()->HasAuthority())
25 {
26 return false;
27 }
28
29 if (!OwnerASC || !SkillTable)
30 {
31 UE_LOG(LogTemp, Error, TEXT("AddSkill Failed: OwnerASC or MasterSkillDataTable is null."));
32 return false;
33 }
34 const FString ContextString(TEXT("Finding Master Skill Data"));
35 //ID로 개별 스킬의 테이블 획득
36 FSkillData* SkillInfo = SkillTable->FindRow<FSkillData>(SkillID, ContextString);
37 if (OwnedSkillHandles.Contains(SkillID))
38 {
39 //이미 보유중인 스킬
40 return true;
41 }
42 else
43 {
44 //신규 스킬 추가
45 if (!SkillInfo || !SkillInfo->AbilityClass)
46 {
47 UE_LOG(LogTemp, Error, TEXT("SkillID '%s' not found or has null AbilityClass in MasterSkillDataTable."), *SkillID.ToString());
48 return false;
49 }
50 // GameplayAbilitySpec을 생성하여 ASC에 어빌리티를 부여합니다.
51 // 스킬 레벨은 1로 시작, InputID는 패시브 스킬이므로 -1(없음)으로 설정합니다.
52 FGameplayAbilitySpec AbilitySpec(SkillInfo->AbilityClass, 1, -1, this);
53 FGameplayAbilitySpecHandle NewHandle = OwnerASC->GiveAbility(AbilitySpec);
54 if (NewHandle.IsValid())
55 {
56 // 성공적으로 부여되었다면, 관리 목록에 추가합니다.
57 OwnedSkillHandles.Add(SkillID, NewHandle);
58 OwnedSkills.Add(SkillID);
59 BroadcastSkillChanged();
60
61 UE_LOG(LogTemp, Log, TEXT("Skill '%s' successfully added."), *SkillID.ToString());
62 InitializeGivenAbility(SkillInfo, NewHandle);
63
64 //StunTags.Add(FGameplayTag::RequestGameplayTag(FName("SkillData.Cooldown.FireTest")));
65 StunTags.Add(SkillInfo->CooldownTag);
66 StunTagEventHandles.Add(OwnerASC->RegisterGameplayTagEvent(
67 StunTags.Last(),
68 EGameplayTagEventType::AnyCountChange
69 ).AddUObject(this, &USkillManagerComponent::OnStunTagChanged)
70 );
71
72 FGameplayTag tmpTag = FGameplayTag::RequestGameplayTag(FName("State.Character.SkillDisabled"),false);
73 StunTagEventHandles.Add(OwnerASC->RegisterGameplayTagEvent(
74 tmpTag,
75 EGameplayTagEventType::AnyCountChange
76 ).AddUObject(this, &USkillManagerComponent::OnStunTagChanged)
77 );
78
79 //broadcast
80 //USkillBuffSystemStatics::ApplyBuff(OwnerASC,FName("SkillData.Active.FIre"),FName("Test.Buff"));
81 return true;
82 }
83 }
84
85 return false;
86}

177void USkillManagerComponent::InitializeGivenAbility(FSkillData* SkillInfo, const FGameplayAbilitySpecHandle& Handle)
178{
179
180 UGameplayAbility* InstancedAbility = OwnerASC->FindAbilitySpecFromHandle(Handle)->GetAbilityInstances()[0];
181 UBaseSkillAbility* AbilityInstance = Cast<UBaseSkillAbility>(InstancedAbility);
182 if (AbilityInstance)
183 {
184 // 인스턴스가 유효하면, InitializeAbility 함수를 호출합니다.
185 AbilityInstance->InitializeAbilityTable(*SkillInfo);
186 AbilityInstance->CalculateFinalStatus();
187 UE_LOG(LogTemp, Log, TEXT("Ability '%s' initialized/updated."), *AbilityInstance->GetName());
188 OwnerASC->TryActivateAbility(Handle);
189 }
