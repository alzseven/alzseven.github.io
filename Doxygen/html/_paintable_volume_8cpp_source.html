35void APaintableVolume::BeginPlay()
36{
37 Super::BeginPlay();
38
39 const float Radius = MissionRadiusSphere->GetScaledSphereRadius();
40 FloorMesh->SetRelativeScale3D(FVector(Radius / 50.f, Radius / 50.f, 1.f));
41
42 if (PaintableGroundMaterial && PaintRenderTargetAsset && PurifyBrushMaterial)
43 {
44 DynamicPaintableMaterial = UMaterialInstanceDynamic::Create(PaintableGroundMaterial, this);
45 BrushMID = UMaterialInstanceDynamic::Create(PurifyBrushMaterial, this);
46
47 if (DynamicPaintableMaterial && BrushMID)
48 {
49 DynamicPaintableMaterial->SetTextureParameterValue(FName("Tex_PaintMask"), PaintRenderTargetAsset);
50 FloorMesh->SetMaterial(0, DynamicPaintableMaterial);
51 }
52 }
53
54 if (HasAuthority())
55 {
56 const int32 TotalCells = GridSize.X * GridSize.Y;
57 GridState.Init(false, TotalCells);
58 ValidGridCells.Init(false, TotalCells);
59 TotalValidGridCells = 0;
60
61 bool bMaskApplied = false;
62 if (ContaminationMask && ContaminationMask->GetPlatformData() && ContaminationMask->GetPlatformData()->Mips.Num() > 0)
63 {
64 FTexture2DMipMap& Mip = ContaminationMask->GetPlatformData()->Mips[0];
65 const void* RawImageData = Mip.BulkData.LockReadOnly();
66 if(RawImageData)
67 {
68 const FColor* ImageData = static_cast<const FColor*>(RawImageData);
69 if (ImageData)
70 {
71 const int32 TextureWidth = Mip.SizeX;
72 const int32 TextureHeight = Mip.SizeY;
73 for (int32 Index = 0; Index < TotalCells; ++Index)
74 {
75 const int32 GridX = Index % GridSize.X;
76 const int32 GridY = Index / GridSize.X;
77 const float U = (float)GridX / (float)(GridSize.X - 1);
78 const float V = (float)GridY / (float)(GridSize.Y - 1);
79 const int32 TexX = FMath::FloorToInt(U * (TextureWidth - 1));
80 const int32 TexY = FMath::FloorToInt(V * (TextureHeight - 1));
81 const FColor PixelColor = ImageData[TexY * TextureWidth + TexX];
82
83 if ((PixelColor.R / 255.0f) >= MaskThreshold)
84 {
85 ValidGridCells[Index] = true;
86 GridState[Index] = true; // 오염된 상태는 true
87 TotalValidGridCells++;
88 }
89 }
90 bMaskApplied = true;
91 }
92 Mip.BulkData.Unlock();
93 }
94 }
95
96 if(!bMaskApplied)
97 {
98 for (int32 Index = 0; Index < TotalCells; ++Index)
99 {
100 if (FVector::DistSquared2D(GridIndexToWorldLocation(Index), GetActorLocation()) < FMath::Square(Radius))
101 {
102 ValidGridCells[Index] = true;
103 GridState[Index] = true;
104 TotalValidGridCells++;
105 }
106 }
107 }
108 }
109}

111void APaintableVolume::Tick(float DeltaSeconds)
112{
113 Super::Tick(DeltaSeconds);
114
115 if (!HasAuthority()) return;
116
117 PurifyTimer -= DeltaSeconds;
118 if (PurifyTimer > 0.f) return;
119
120 bool bStateChanged = false;
121
122 for (FConstPlayerControllerIterator It = GetWorld()->GetPlayerControllerIterator(); It; ++It)
123 {
124 ACharacter* PlayerCharacter = It->Get() ? It->Get()->GetPawn<ACharacter>() : nullptr;
125 if (PlayerCharacter)
126 {
127 const FVector PlayerLocation = PlayerCharacter->GetActorLocation();
128 for (int32 Index = 0; Index < GridState.Num(); ++Index)
129 {
130 if (ValidGridCells[Index] && GridState[Index] == true)
131 {
132 if (FVector::DistSquared2D(PlayerLocation, GridIndexToWorldLocation(Index)) < FMath::Square(PlayerPurifyRadius))
133 {
134 GridState[Index] = false; // 정화되면 false
135 bStateChanged = true;
136 }
137 }
138 }
139 }
140 }
141
142 if (bStateChanged)
143 {
144 PurifyTimer = PurifyCooldown;
145 OnStateUpdated.Broadcast();
146 }
147}

149void APaintableVolume::OnRep_GridState()
150{
151 // VisualizedGridState가 아직 초기화되지 않았다면, 최초의 상태 동기화입니다.
152 if (VisualizedGridState.Num() != GridState.Num())
153 {
154 // [핵심 수정 1] 클라이언트의 시각화 상태 배열을 초기화합니다.
155 VisualizedGridState.Init(false, GridState.Num());
156
157 // 1. 렌더 타겟을 깨끗한 상태(흰색)로 완전히 초기화합니다.
158 // 정화 브러시가 흰색을 칠하므로, 오염된 부분은 검은색이어야 합니다.
159 UKismetRenderingLibrary::ClearRenderTarget2D(this, PaintRenderTargetAsset, FLinearColor::White);
160
161 // 2. 오염 마스크를 렌더 타겟에 그려줄 전용 머티리얼이 설정되어 있는지 확인합니다.
162 if (ContaminationStampMaterial && ContaminationMask)
163 {
164 // 스탬프 머티리얼의 동적 인스턴스를 생성합니다.
165 UMaterialInstanceDynamic* StampMID = UMaterialInstanceDynamic::Create(ContaminationStampMaterial, this);
166 if (StampMID)
167 {
168 // 머티리얼의 텍스처 파라미터에 오염 마스크 텍스처를 설정합니다.
169 // (머티리얼 내 텍스처 파라미터의 이름은 "MaskTexture" 또는 다른 이름일 수 있습니다. 일치시켜야 합니다.)
170 StampMID->SetTextureParameterValue(FName("Texture"), ContaminationMask);
171
172 // 머티리얼을 렌더 타겟 전체에 그려서 초기 오염 상태를 만듭니다.
173 UKismetRenderingLibrary::DrawMaterialToRenderTarget(this, PaintRenderTargetAsset, StampMID);
174 }
175 }
176
177 // 3. 서버의 논리적 상태를 클라이언트의 시각적 상태 추적 배열에 복사합니다.
178 for (int32 Index = 0; Index < GridState.Num(); ++Index)
179 {
180 VisualizedGridState[Index] = GridState[Index];
181 }
182 }
183 else // 최초 동기화 이후, 상태가 업데이트될 때마다 호출됩니다.
184 {
185 // [핵심 수정 2] 새로 정화된(true -> false) 칸만 찾아서 그립니다.
186 for (int32 Index = 0; Index < GridState.Num(); ++Index)
187 {
188 // 서버에서는 정화됐는데(false), 아직 클라이언트는 오염된 것으로 안다면(true)
189 if (GridState.IsValidIndex(Index) && VisualizedGridState.IsValidIndex(Index) &&
190 GridState[Index] == false && VisualizedGridState[Index] == true)
191 {
192 // 클라이언트의 시각화 상태도 정화된 것으로 업데이트합니다.
193 VisualizedGridState[Index] = false;
194 // 해당 위치에 정화 브러시(흰색)를 그립니다.
195 DrawAtLocation(GridIndexToWorldLocation(Index));
196 }
197 }
198 }

201void APaintableVolume::DrawAtLocation(const FVector& WorldLocation)
202{
203 if (!PaintRenderTargetAsset || !BrushMID) return;
204
205 const float Radius = MissionRadiusSphere->GetScaledSphereRadius();
206 const FVector Origin = GetActorLocation();
207
208 const FVector2D DrawUV = FVector2D(
209 (WorldLocation.X - (Origin.X - Radius)) / (Radius * 2.f),
210 (WorldLocation.Y - (Origin.Y - Radius)) / (Radius * 2.f)
211 );
212
213 BrushMID->SetVectorParameterValue(FName("DrawLocation"), FLinearColor(DrawUV.X, DrawUV.Y, 0, 0));
214 BrushMID->SetScalarParameterValue(FName("BrushSize"), PlayerPurifyRadius / (Radius * 2.f));
215
216 UKismetRenderingLibrary::DrawMaterialToRenderTarget(this, PaintRenderTargetAsset, BrushMID);
217}

219FVector APaintableVolume::GridIndexToWorldLocation(int32 Index) const
220{
221 const float Radius = MissionRadiusSphere->GetScaledSphereRadius();
222 const FVector Origin = GetActorLocation();
223 const FVector BoundingBoxMin = Origin - FVector(Radius, Radius, 0);
224
225 const FVector CellSize = FVector(Radius * 2 / GridSize.X, Radius * 2 / GridSize.Y, 1.f);
226
227 const int32 X = Index % GridSize.X;
228 const int32 Y = Index / GridSize.X;
229
230 return BoundingBoxMin + FVector(X * CellSize.X + CellSize.X / 2, Y * CellSize.Y + CellSize.Y / 2, Origin.Z);
231}

233int32 APaintableVolume::GetRemainingPaintedCells() const
234{
235 int32 Count = 0;
236 // 이 함수는 서버에서만 호출되므로 HasAuthority 체크는 불필요할 수 있지만, 안전을 위해 유지합니다.
237 if(HasAuthority())
238 {
239 for(int32 i=0; i < GridState.Num(); ++i)
240 {
241 if(ValidGridCells[i] && GridState[i])
242 {
243 Count++;
244 }
245 }
246 }
247 else // 클라이언트에서도 정확한 계산이 필요하다면 VisualizedGridState를 사용합니다.
248 {
249 for(int32 i=0; i < VisualizedGridState.Num(); ++i)
250 {
251 // 클라이언트에는 ValidGridCells가 없으므로 모든 셀을 대상으로 계산하거나,
252 // ValidGridCells도 복제해야 합니다. 지금은 GridState에 의존하는 서버 로직을 따릅니다.
253 if(GridState.IsValidIndex(i) && ValidGridCells.IsValidIndex(i) && ValidGridCells[i] && GridState[i])
254 {
255 Count++;
256 }
257 }
258 }
259 return Count;
260}
