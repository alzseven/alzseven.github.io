13void ULevelSelect::NativeConstruct()
14{
15 Super::NativeConstruct();
16
17 FAssetRegistryModule& AssetRegistryModule = FModuleManager::LoadModuleChecked<FAssetRegistryModule>(TEXT("AssetRegistry"));
18 IAssetRegistry& AssetRegistry = AssetRegistryModule.Get();
19
20 // 1. 에셋 레지스트리가 이미 준비되었다면 즉시 목록을 채웁니다.
21 if (!AssetRegistry.IsLoadingAssets())
22 {
23 PopulateLevelList();
24 }
25 // 2. 아직 로딩 중이라면, 로딩이 끝났을 때 PopulateLevelList 함수가 호출되도록 예약(바인딩)합니다.
26 else
27 {
28 OnFilesLoadedHandle = AssetRegistry.OnFilesLoaded().AddUObject(this, &ULevelSelect::PopulateLevelList);
29 }
30}

42void ULevelSelect::PopulateLevelList()
43{
44 // 1. 유효성 검사: 필요한 위젯과 클래스가 설정되었는지 확인
45 if (!LevelListContainer || !LevelItemWidgetClass)
46 {
47 return;
48 }
49
50 // 2. 목록 초기화: 기존에 있던 항목들을 모두 제거
51 LevelListContainer->ClearChildren();
52
53 // 3. 모든 레벨 에셋 가져오기
54 const TArray<TSoftObjectPtr<UWorld>> AllLevels = UGetLevelFunctionLibrary::GetAllLevelAssets();
55
56 // 4. 각 레벨에 대해 항목 위젯 생성 및 추가
57 for (const TSoftObjectPtr<UWorld>& LevelAsset : AllLevels)
58 {
59 // LevelItemWidgetClass로 위젯 인스턴스 생성
60 ULevelSelectItem* ItemWidget = CreateWidget<ULevelSelectItem>(this, LevelItemWidgetClass);
61 if (ItemWidget)
62 {
63 // 생성된 위젯에 레벨 정보 전달 (Setup 함수 호출)
64 ItemWidget->LevelDataSetup(LevelAsset);
65 // 컨테이너에 자식으로 추가하여 화면에 표시
66 LevelListContainer->AddChild(ItemWidget);
67 }
68 }
69}
