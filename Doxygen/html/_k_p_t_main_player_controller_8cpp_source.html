130void AKPTMainPlayerController::TryInitializeUI()
131{
132 // 이미 초기화되었거나, 로컬 플레이어가 아니거나, PlayerState가 아직 유효하지 않으면 실행하지 않습니다.
133 if (bHasInitializedUI || !IsLocalController() || GetPlayerState<APlayerState>() == nullptr)
134 {
135 return;
136 }
137
138 bHasInitializedUI = true;
139
140 if (ULocalPlayer* LocalPlayer = GetLocalPlayer())
141 {
142 if (UUIManagerSubsystem* UIManager = LocalPlayer->GetSubsystem<UUIManagerSubsystem>())
143 {
144 UIManager->SetupInitialUI();
145 }
146 }
147
148 OnPlayerStateReady.Broadcast(this, GetPlayerState<APlayerState>());
149 GetPlayerState<AKPTPlayerState>()->BroadCastStateTag();
150 // 블루프린트에서 UI 생성 및 바인딩을 처리하도록 이벤트를 호출합니다.
151 Client_InitializeHUD();
152}

173void AKPTMainPlayerController::AbilityInputTagHeld(FGameplayTag InputTag)
174{
175 if (!InputTag.IsValid()) return;
176 if (InputTag.MatchesTagExact(FGameplayTag::RequestGameplayTag(FName("InputTag.SpaceBar"))))
177 {
178 // --- 대쉬 어빌리티를 위한 GameplayEvent 로직 ---
179 APawn* ControlledPawn = GetPawn();
180 if (!ControlledPawn) return;
181
182 // 1. 방향 데이터 계산
183 FVector DashDirection = ControlledPawn->GetLastMovementInputVector();
184 if (DashDirection.IsNearlyZero())
185 {
186 DashDirection = ControlledPawn->GetActorForwardVector();
187 }
188 FRotator DashRotation = DashDirection.Rotation();
189
190 // 2. GameplayEventData 생성 및 데이터 패킹
191 FGameplayEventData Payload;
192 Payload.EventTag = FGameplayTag::RequestGameplayTag(FName("Event.Ability.Dash")); // 어빌리티를 트리거할 고유 이벤트 태그
193
194 FGameplayAbilityTargetDataHandle TargetDataHandle;
195 FGameplayAbilityTargetData_LocationInfo* LocationData = new FGameplayAbilityTargetData_LocationInfo();
196 LocationData->SourceLocation.LocationType = EGameplayAbilityTargetingLocationType::ActorTransform;
197 LocationData->SourceLocation.SourceActor = ControlledPawn;
198 LocationData->TargetLocation.LocationType = EGameplayAbilityTargetingLocationType::LiteralTransform;
199 // TargetLocation의 Transform에 방향 정보를 저장 (위치는 방향, 회전은 0)
200 LocationData->TargetLocation.LiteralTransform.SetLocation(ControlledPawn->GetActorLocation());
201 LocationData->TargetLocation.LiteralTransform.SetScale3D(DashDirection);
202 TargetDataHandle.Add(LocationData);
203 Payload.TargetData = TargetDataHandle;
204
205 const FString RoleString = ControlledPawn->HasAuthority() ? TEXT("Server") : TEXT("Client");
206 UE_LOG(LogTemp, Warning, TEXT("[%s] TargetRotation: %s"), *RoleString, *DashDirection.ToString());
207
208 // 3. Pawn에게 이벤트 전송
209 UAbilitySystemBlueprintLibrary::SendGameplayEventToActor(ControlledPawn, Payload.EventTag, Payload);
210 }
211 else
212 {
213 if (UKPTAbilitySystemComponent* ASC = GetASC())
214 {
215 ASC->AbilityInputTagHeld(InputTag);
216 }
217 }
218}

264void AKPTMainPlayerController::SetupInputComponent()
265{
266 Super::SetupInputComponent();
267
268 UEnhancedInputComponent* EnhancedInputComponent = CastChecked<UEnhancedInputComponent>(InputComponent);
269 check(InputConfig);
270 TArray<FKPTInputAction>& InputActions = InputConfig->ActiveActions;
271 for (const FKPTInputAction& Action : InputActions)
272 {
273 if (Action.InputAction && Action.InputTag.IsValid())
274 {
275 // 변경 후: ASC의 내장 함수를 호출하는 새로운 전달 함수에 연결한다.
276 EnhancedInputComponent->BindAction(Action.InputAction, ETriggerEvent::Started, this, &AKPTMainPlayerController::AbilityInputTagPressed, Action.InputTag);
277 EnhancedInputComponent->BindAction(Action.InputAction, ETriggerEvent::Triggered, this, &AKPTMainPlayerController::AbilityInputTagHeld, Action.InputTag);
278 EnhancedInputComponent->BindAction(Action.InputAction, ETriggerEvent::Completed, this, &AKPTMainPlayerController::AbilityInputTagReleased, Action.InputTag);
279 }
280 }
