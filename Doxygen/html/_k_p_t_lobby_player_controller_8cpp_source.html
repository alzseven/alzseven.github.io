17void AKPTLobbyPlayerController::BeginPlay()
18{
19 Super::BeginPlay();
20
21 if (IsLocalController())
22 {
23 CreateLobbyWidget();
24
25 // --- 여기가 핵심 변경점 ---
26 // 클라이언트로 카메라 액터가 복제될 때까지 0.1초마다 확인하는 타이머를 시작합니다.
27 GetWorldTimerManager().SetTimer(
28 SetCameraTimerHandle,
29 this,
30 &AKPTLobbyPlayerController::TryToSetLobbyCamera,
31 0.1f,
32 true);
33 }
34}

39 GetWorldTimerManager().ClearTimer(SetCameraTimerHandle);

43void AKPTLobbyPlayerController::TryToSetLobbyCamera()
44{
45 // "LobbyCamera" 태그를 가진 액터를 찾습니다.
46 TArray<AActor*> FoundActors;
47 UGameplayStatics::GetAllActorsWithTag(this, LobbyCameraTag, FoundActors);
48
49 // 카메라 액터를 찾았다면
50 if (FoundActors.Num() > 0)
51 {
52 // 뷰 타겟을 설정하고,
53 SetViewTarget(FoundActors[0]);
54
55 // --- 여기가 핵심 변경점 ---
56 // 성공했으므로 더 이상 타이머를 실행할 필요가 없습니다. 타이머를 즉시 중지합니다.
57 GetWorldTimerManager().ClearTimer(SetCameraTimerHandle);
58 }
59}

101void AKPTLobbyPlayerController::Server_SelectCharacter_Implementation(FGameplayTag CharacterID)
102{
103 if (AKPTPlayerState* PS = GetPlayerState<AKPTPlayerState>())
104 {
105 // 1. PlayerState 컴포넌트에 먼저 상태를 저장합니다 (권위 있는 원본 데이터).
106 PS->GetLobbyStateComponent()->SetSelectedCharacter(CharacterID);
107
108 // 2. GameMode를 통해 GameState의 정보를 업데이트하도록 요청합니다.
109 if (AKPTLobbyGameMode* GM = GetWorld()->GetAuthGameMode<AKPTLobbyGameMode>())
110 {
111 GM->UpdatePlayerSelectedCharacter(this, CharacterID);
112 }
113 }
