20void UActiveAbility_Slash::CalculateFinalStatus()
21{
22 FDetailedSkillData StackedStatusData;
23 StackedStatusData.AttackArea = 0.f;
24 StackedStatusData.CooldownTime = 0.f;
25 StackedStatusData.MaxPierceCount = 0.f;
26 StackedStatusData.NumToSpawn = 0.f;
27 StackedStatusData.ProjectileSizeMultiplier = 0.f;
28 StackedStatusData.ProjectileSpeed = 0.f;
29 StackedStatusData.SkillDamage = 0.f;
30 StackedStatusData.ProjectileLife = 0.f;
31
32 UAbilitySystemComponent* ASC = GetAbilitySystemComponentFromActorInfo();
33 if (!ASC) return;
34
35 FGameplayTagContainer QueryTags;
36 QueryTags.AddTag(SkillTag);
37
38 TArray<FActiveGameplayEffectHandle> FoundEffects = ASC->GetActiveEffectsWithAllTags(QueryTags);
39 for (FActiveGameplayEffectHandle& Handle : FoundEffects)
40 {
41 const FGameplayEffectSpec& Spec = ASC->GetActiveGameplayEffect(Handle)->Spec;
42 float Magnitude = 0.f;
43
44 Magnitude = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.ProjectileSpeed")), false, -100.f);
45 if (Magnitude != -100)
46 {
47 StackedStatusData.ProjectileSpeed += Magnitude;
48 }
49 Magnitude = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.ProjectileSize")), false, -100.f);
50 if (Magnitude != -100)
51 {
52 StackedStatusData.ProjectileSizeMultiplier += Magnitude;
53 }
54 Magnitude = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.Cooldown")), false, -100.f);
55 if (Magnitude != -100)
56 {
57 StackedStatusData.CooldownTime += Magnitude;
58 }
59 Magnitude = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.AttackArea")), false, -100.f);
60 if (Magnitude != -100)
61 {
62 StackedStatusData.AttackArea += Magnitude;
63 }
64 Magnitude = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.Damage")), false, -100.f);
65 if (Magnitude != -100)
66 {
67 StackedStatusData.SkillDamage += Magnitude;
68 }
69 Magnitude = Spec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Modifiers.Pierce")), false, -100.f);
70 if (Magnitude != -100)
71 {
72 if (Magnitude == -1 || SkillSpawnData.MaxPierceCount == -1) StackedStatusData.MaxPierceCount = -1;
73 else StackedStatusData.MaxPierceCount += Magnitude;
74 }
75 }
76 SkillSpawnData.NumToSpawn += StackedStatusData.NumToSpawn;
77 SkillSpawnData.ProjectileSpeed *= FMath::Max(0.f,1.f + StackedStatusData.ProjectileSpeed);
78 SkillSpawnData.ProjectileSizeMultiplier *= FMath::Max(0.f, 1.f + StackedStatusData.ProjectileSizeMultiplier);
79 SkillSpawnData.CooldownTime *= FMath::Max(0.f,1.f + StackedStatusData.CooldownTime);
80 SkillSpawnData.AttackArea *= FMath::Max(0.f,1.f + StackedStatusData.AttackArea);
81 SkillSpawnData.SkillDamage *= FMath::Max(0.f,1.f + StackedStatusData.SkillDamage);
82 if (SkillSpawnData.MaxPierceCount == -1 || StackedStatusData.MaxPierceCount == -1)
83 SkillSpawnData.MaxPierceCount = -1;
84 else SkillSpawnData.MaxPierceCount += StackedStatusData.MaxPierceCount;
85}

87void UActiveAbility_Slash::SpawnProjectile()
88{
89 bool bIsServer = GetCurrentActorInfo()->AvatarActor->HasAuthority();
90 if (!bIsServer) return;
91
92 if (!SlashProjectileClass)
93 {
94 UE_LOG(LogTemp, Error, TEXT("SlashProjectileClass is not set"));
95 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, true);
96 return;
97 }
98
99 AActor* AvatarActor = GetAvatarActorFromActorInfo();
100 if (!AvatarActor)
101 {
102 UE_LOG(LogTemp, Error, TEXT("AvatarActor is not set"));
103 EndAbility(GetCurrentAbilitySpecHandle(), GetCurrentActorInfo(), GetCurrentActivationInfo(), true, true);
104 return;
105 }
106
107 FTransform SpawnTransform = AvatarActor->GetActorTransform();
108
109 UWorld* World = GetWorld();
110 if (World)
111 {
112
113
114 UProjectilePoolSubsystem* PoolSubsystem = GetWorld()->GetSubsystem<UProjectilePoolSubsystem>();
115 if (!PoolSubsystem)
116 {
117 UE_LOG(LogTemp, Error, TEXT("UProjectilePoolSubsystem not found"));
118 return;
119 }
120
121 AProjectileBase* SpawnedSlash = PoolSubsystem->RequestProjectile(SlashProjectileClass);
122 if (!SpawnedSlash)
123 {
124 return;
125 }
126
127 SpawnedSlash->PayloadEffectClass = PayloadEffectClass;
128 CalculateFinalStatus();
129 if (UProjectileMovementComponent* ProjectileMovementComp = SpawnedSlash->FindComponentByClass<UProjectileMovementComponent>())
130 {
131 ProjectileMovementComp->InitialSpeed = SkillSpawnData.ProjectileSpeed;
132 ProjectileMovementComp->MaxSpeed = SkillSpawnData.ProjectileSpeed;
133 }
134 SpawnedSlash->ActivateProjectile(
135 SkillSpawnData,
136 VFXCueTag,
137 GetAbilitySystemComponentFromActorInfo(),
138 SpawnTransform,
139 GetOwningActorFromActorInfo(),
140 Cast<APawn>(GetAvatarActorFromActorInfo()),
141 0,1
142 );
143 }
144}
