294void AMinionMonster::FinishDying()
295{
296 SetActorHiddenInGame(true);
297
298 TArray<AActor*> AttachedActors;
299 GetAttachedActors(AttachedActors);
300
301 for (AActor* DamageFloater : AttachedActors)
302 {
303 DamageFloater->DetachFromActor(FDetachmentTransformRules(EDetachmentRule::KeepWorld, true));
304 DamageFloater->Destroy();
305 }
306
307 if (HasAuthority())
308 {
309 for (const FGameplayAbilitySpec& Spec : GetAbilitySystemComponent()->GetActivatableAbilities())
310 {
311 if (!IsValid(AttackAbilityClass))
312 break;
313
314 if (Spec.Ability && Spec.Ability->GetClass() == AttackAbilityClass)
315 {
316 if (Spec.IsActive())
317 {
318 GetAbilitySystemComponent()->CancelAbilityHandle(Spec.Handle);
319 }
320 }
321 }
322
323 UPoolSubSystem* PoolSubSystem = GetGameInstance()->GetSubsystem<UPoolSubSystem>();
324
325 check(PoolSubSystem);
326
327 PoolSubSystem->ReturnToPool(this);
328
329 }

450void AMinionMonster::FindTarget()
451{
452 AGameStateBase* GS = GetWorld()->GetGameState();
453 TArray<AActor*> OutActor;
454
455 for (int i = 0; i < GS->PlayerArray.Num(); ++i)
456 {
457 IAbilitySystemInterface* ASCInterface = Cast<IAbilitySystemInterface>(GS->PlayerArray[i]);
458 if (ASCInterface)
459 {
460 UAbilitySystemComponent* ASC = ASCInterface->GetAbilitySystemComponent();
461 if (ASC->HasMatchingGameplayTag(FGameplayTag::RequestGameplayTag(FName("State.Character.Dead"))))
462 {
463 continue;
464 }
465 }
466
467 OutActor.Add(GS->PlayerArray[i]->GetPawn());
468 }
469
470 if (OutActor.Num() <= 0)
471 {
472 SetActorTickEnabled(false);
473 UnRegisterToFlockingManager();
474 return;
475 }
476
477 float MaxDistance = MAX_FLT;
478
479 AActor* TargetActor = nullptr;
480 for (auto& PlayerActor : OutActor)
481 {
482 check(PlayerActor);
483 float Distance = (PlayerActor->GetActorLocation() - GetActorLocation()).Length();
484 if (Distance < MaxDistance)
485 {
486 MaxDistance = Distance;
487 TargetActor = PlayerActor;
488 }
489 }
490
491 FlockingComponent->SetTarget(TargetActor);
492
493 if (HasAuthority() && (TargetActor != nullptr))
494 {
495 if (APlayerCharacter* Player = Cast<APlayerCharacter>(TargetActor))
496 {
497 Player->OnPlayerDied.RemoveDynamic(this, &AMinionMonster::OnTargetDied);
498 Player->OnPlayerDied.AddDynamic(this, &AMinionMonster::OnTargetDied);
499 }
500 }
