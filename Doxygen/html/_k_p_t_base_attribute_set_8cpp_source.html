63void UKPTBaseAttributeSet::PostGameplayEffectExecute(const FGameplayEffectModCallbackData& Data)
64{
65 Super::PostGameplayEffectExecute(Data);
66
67 FGameplayEffectContextHandle Context = Data.EffectSpec.GetContext();
68 UAbilitySystemComponent* SourceASC = Context.GetOriginalInstigatorAbilitySystemComponent();
69
70 AActor* SourceActor = SourceASC ? SourceASC->GetAvatarActor_Direct() : nullptr;
71 AActor* TargetActor = nullptr;
72
73 ICombatInterface* TargetCombatInterface = nullptr;
74
75 // Target 이벤트 처리를 위한 TargetActor 및 CombatInterface 세팅
76 if (Data.Target.AbilityActorInfo.IsValid() && Data.Target.AbilityActorInfo->AvatarActor.IsValid())
77 {
78 TargetActor = Data.Target.AbilityActorInfo->AvatarActor.Get();
79
80 if (TargetActor)
81 {
82 TargetCombatInterface = Cast<ICombatInterface>(TargetActor);
83 }
84 }
85
86 // Get the Target Tags from the Effect Spec
87 if (Data.EvaluatedData.Attribute == GetDamageAttribute())
88 {
89 // 치명타 여부
90 // TODO: Handle in case of cannot-crit
91 bool isCrit = Data.EffectSpec.GetSetByCallerMagnitude(
92 FGameplayTag::RequestGameplayTag(FName("Data.Damage.IsCritical")), false, 0.f) > 0.f;
93 //bool isCrit = Data.EffectSpec.GetSetByCallerMagnitude(FGameplayTag::RequestGameplayTag(FName("SkillData.Active.Dash")), false, 0.f) > 0.f;
94
95 const float LocalDamageDone = GetDamage();
96 SetDamage(0.f); // Clear the meta attribute
97
98 if (LocalDamageDone > 0.f && TargetActor)
99 {
100 const float OldHealth = GetHealth();
101 const float NewHealth = FMath::Clamp(OldHealth - LocalDamageDone, 0.0f, GetMaxHealth());
102 SetHealth(NewHealth);
103 if (ABossMonsterBase* Boss = Cast<ABossMonsterBase>(TargetActor))
104 {
105 Boss->AddStackedDamage(LocalDamageDone);
106 }
107 FGameplayEventData HitPayload;
108 HitPayload.EventTag = FGameplayTag::RequestGameplayTag(FName("GameplayFlow.OnHit"));
109
110 HitPayload.Instigator = SourceActor;
111 HitPayload.Target = TargetActor;
112 if (SourceASC)
113 {
114 SourceASC->HandleGameplayEvent(HitPayload.EventTag, &HitPayload);
115 }
116
117 // *** 서버 UI 업데이트를 위한 핵심 수정 부분 ***
118 // 로컬 컨트롤 대상(서버 또는 싱글 플레이어)이고, 실제로 체력이 변했다면
119 if (Data.Target.AbilityActorInfo.IsValid() && Data.Target.AbilityActorInfo->IsLocallyControlled() &&
120 OldHealth != NewHealth)
121 {
122 // RepNotify가 클라이언트에서만 자동 호출되므로, 서버에서는 수동으로 호출해줍니다.
123 OnRep_Health(FGameplayAttributeData(OldHealth));
124 }
125
126 if (TargetCombatInterface)
127 {
128 TargetCombatInterface->Damaged(); //이거 중복 처리해야 함.
129 //test
130 }
131
132 //Call DamagetFloater
133 if (SourceActor)
134 {
135 // 플레이어가 공격시
136 AKPTMainPlayerController* AttackerController = Cast<AKPTMainPlayerController>(
137 SourceActor->GetInstigatorController());
138
139 if (AttackerController)
140 {
141 AttackerController->ShowDamageFloater(LocalDamageDone, isCrit, TargetActor);
142 }
143 else
144 {
145 // 피격 시
146 AttackerController = Cast<AKPTMainPlayerController>(TargetActor->GetInstigatorController());
147
148 if (AttackerController)
149 {
150 AttackerController->ShowDamageFloater(-LocalDamageDone, isCrit, TargetActor);
151 }
152 }
153 }
154
155 // 사망 처리 (현재 Attribute는 Target )
156 if (GetHealth() <= 0.0f)
157 {
158 // 사망 처리 로직 호출 (캐릭터 클래스에 구현)
159
160 // Target이 죽었다면 Source와 Target두명 다 이벤트를 보내야한다.
161 FGameplayEventData DeathPayload;
162 DeathPayload.EventTag = FGameplayTag::RequestGameplayTag(FName("Event.Character.Death"));
163 DeathPayload.Instigator = SourceActor;
164 DeathPayload.Target = TargetActor;
165 Data.Target.HandleGameplayEvent(DeathPayload.EventTag, &DeathPayload);
166
167 FGameplayEventData KillPayload;
168 KillPayload.EventTag = FGameplayTag::RequestGameplayTag(FName("GameplayFlow.OnKill"));
169 KillPayload.Instigator = SourceActor;
170 KillPayload.Target = TargetActor;
171 SourceASC->HandleGameplayEvent(KillPayload.EventTag, &KillPayload);
172
173 if (TargetCombatInterface)
174 {
175 TargetCombatInterface->Die();
176 }
177 }
178 }
179 }
180 else if (Data.EvaluatedData.Attribute == GetHealthAttribute())
181 {
182 SetHealth(FMath::Clamp(GetHealth(), 0.0f, GetMaxHealth()));
183 }
184 else if (Data.EvaluatedData.Attribute == GetMaxHealthAttribute()) // 최대 체력이 변경된 경우
185 {
186 AdjustAttributeForMaxChange(Health, MaxHealth, GetMaxHealth(), GetHealthAttribute());
187 }
188}

198// 최대치 변경 시 현재 값 비율 유지 (예: 최대 체력 증가 시 현재 체력도 비율에 맞게 증가)
199void UKPTBaseAttributeSet::AdjustAttributeForMaxChange(FGameplayAttributeData& AffectedAttribute,
200 const FGameplayAttributeData& MaxAttribute, float NewMaxValue,
201 const FGameplayAttribute& AffectedAttributeProperty)
202{
203 UAbilitySystemComponent* ASC = GetOwningAbilitySystemComponent();
204 const float CurrentMaxValue = MaxAttribute.GetCurrentValue(); // 이전 최대값
205 if (!FMath::IsNearlyEqual(CurrentMaxValue, NewMaxValue) && ASC && NewMaxValue > 0 && CurrentMaxValue > 0)
206 // 0으로 나누기 방지
207 {
208 const float CurrentValue = AffectedAttribute.GetCurrentValue();
209 //float NewDelta = (CurrentValue * NewMaxValue / CurrentMaxValue) - CurrentValue;
210 //ASC->ApplyModToAttributeUnsafe(AffectedAttributeProperty, EGameplayModOp::Additive, NewDelta);
211
212 float NewCurrentValue = (CurrentValue * NewMaxValue / CurrentMaxValue);
213
214 AffectedAttribute.SetBaseValue(NewCurrentValue);
215 AffectedAttribute.SetCurrentValue(NewCurrentValue);
216 }
