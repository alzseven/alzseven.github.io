168void APlayerCharacter::Server_StartRevive_Implementation(TSubclassOf<UGameplayEffect> ReviveEffectClass)
169{
170 AKPTPlayerState* KPT_PS = GetPlayerState<AKPTPlayerState>();
171 if (KPT_PS && KPT_PS->GetAbilitySystemComponent() && DeadStateEffect)
172 {
173 UAbilitySystemComponent* ASC = KPT_PS->GetAbilitySystemComponent();
174
175 // [변경] 1. 사망 태그를 부여한 GE를 제거합니다.
176 FGameplayEffectQuery Query;
177 Query.EffectDefinition = DeadStateEffect;
178 ASC->RemoveActiveEffects(Query); // DeadStateEffect로부터 생성된 모든 이펙트 인스턴스 제거
179
180 // 2. 체력 회복 GE 적용 (이 부분은 기존 로직과 동일)
181 if (ReviveEffectClass)
182 {
183 FGameplayEffectContextHandle ContextHandle = ASC->MakeEffectContext();
184 ASC->ApplyGameplayEffectToSelf(ReviveEffectClass->GetDefaultObject<UGameplayEffect>(), 1.0f, ContextHandle);
185 }
186 }
187
188 // 3. 모든 클라이언트에게 부활 시각 효과를 지시
189 Multicast_FinishRevive();
190
191 // 4. 애니메이션 후 컨트롤러 재빙의 (기존 로직과 동일)
192 FTimerHandle PossessTimerHandle;
193 float DelayBeforePossess = 2.0f;
194
195 GetWorldTimerManager().SetTimer(
196 PossessTimerHandle,
197 [this]()
198 {
199 AController* MyController = GetController();
200 if (MyController)
201 {
202 MyController->Possess(this);
203 }
204 },
205 DelayBeforePossess,
206 false
207 );

272void APlayerCharacter::Tick(float DeltaTime)
273{
274 Super::Tick(DeltaTime);
275 MouseRotate = CalculateMouseRotation();
276 if (IsLocallyControlled())
277 {
278 if (AbilitySystemComponent && !AbilitySystemComponent->HasMatchingGameplayTag(FGameplayTag::RequestGameplayTag(FName("Player.State.UsingActive"))))
279 {
280 APlayerController* PC = Cast<APlayerController>(GetController());
281 if (PC)
282 {
283 FHitResult HitResult;
284 PC->GetHitResultUnderCursor(ECC_Visibility, false, HitResult);
285 if (HitResult.bBlockingHit)
286 {
287 const FVector WorldDirection = (HitResult.ImpactPoint - GetActorLocation()).GetSafeNormal();
288 const FRotator TargetRotation = FRotationMatrix::MakeFromX(WorldDirection).Rotator();
289
290 // Pitch Roll 0으로 고정 -> 캐릭터기울기
291 FRotator FinalRotation(0.f, TargetRotation.Yaw, 0.f);
292
293 FRotator TargetBodyRotation = GetActorRotation();
294 TargetBodyRotation.Yaw = FinalRotation.Yaw; // 목표는 AimRotation의 Yaw값
295 FRotator NewRotation = FMath::RInterpTo(GetActorRotation(), TargetBodyRotation, DeltaTime, 12.0f);
296 SetActorRotation(NewRotation);
297
298 if (!HasAuthority())
299 {
300 Server_UpdateRotation(NewRotation);
301 }
302 }
303 }
304 }
305 }
306
307 // 2D 블렌드 스페이스용 변수
308 UCharacterMovementComponent* MoveComp = GetCharacterMovement();
309 if (MoveComp->GetCurrentAcceleration().SizeSquared() > KINDA_SMALL_NUMBER)
310 {
311 const FVector WorldVelocity = MoveComp->Velocity;
312 const FVector LocalVelocityDirection = GetActorRotation().UnrotateVector(WorldVelocity.GetSafeNormal());
313
314 Forward = LocalVelocityDirection.X;
315 Right = LocalVelocityDirection.Y;
316 }
317 else
318 {
319 Forward = 0.f;
320 Right = 0.f;
321 }
322
323 switch (CurrentViewMode)
324 {
325 case EViewMode::QuarterView:
326 if (KINDA_SMALL_NUMBER < DirectionToMove.SizeSquared())
327 {
328 //GetController()->SetControlRotation(FRotationMatrix::MakeFromX(DirectionToMove).Rotator());
329 AddMovementInput(DirectionToMove);
330 DirectionToMove = FVector::ZeroVector;
331 }
332 break;
333 case EViewMode::None:
334 case EViewMode::End:
335 default:
336 break;
337 }
338}

340FRotator APlayerCharacter::CalculateMouseRotation()
341{
342 APlayerController* PlayerController = Cast<APlayerController>(GetController());
343 if (!PlayerController) return GetActorRotation();
344
345 FHitResult HitResult;
346 if (PlayerController->GetHitResultUnderCursor(ECC_Visibility, true, HitResult))
347 {
348 FVector PlayerLocation = GetActorLocation();
349 FVector MouseLocation = HitResult.Location;
350
351 PlayerLocation.Z = 0.f;
352 MouseLocation.Z = 0.f;
353
354 FRotator LookAtRotation = (MouseLocation - PlayerLocation).Rotation();
355 return FRotator(0.f, LookAtRotation.Yaw, 0.f);
356 }
357 return GetActorRotation();
358}

604void APlayerCharacter::DetectInteractables_Implementation()
605{
606 // UKismetSystemLibrary::PrintString(this,
607 // FString::Printf(TEXT("%d"),PlayerHeldItems.Num()));
608 FVector Start = GetActorLocation();
609 FVector End = Start;
610 float Radius = 300.f;
611
612 TArray<FHitResult> HitResults;
613 FCollisionQueryParams Params;
614 Params.AddIgnoredActor(this);
615
616 bool bHit = GetWorld()->SweepMultiByChannel(
617 HitResults,
618 Start,
619 End,
620 FQuat::Identity,
621 ECollisionChannel::ECC_GameTraceChannel2,
622 FCollisionShape::MakeSphere(Radius),
623 Params
624 );
625 FColor DebugColor = HitResults.Num()>0 ? FColor::Red : FColor::Green;
626
627 if (DebugLineVisible)
628 {
629 DrawDebugSphere(
630 GetWorld(),
631 Start,
632 Radius,
633 12,
634 DebugColor,
635 false,
636 0.1f
637 );
638 }
639
640
641 AActor* SelectActor = nullptr;
642 float tmpDistance = FLT_MAX;
643
644 for (FHitResult& Hit: HitResults)
645 {
646 if (DebugLineVisible)
647 DrawDebugPoint(GetWorld(), Hit.ImpactPoint, 15.0f, FColor::Yellow, false, 0.1f);
648 // UKismetSystemLibrary::PrintString(this,
649 // Hit.GetActor()->GetName());
650 AActor* HitActor = Hit.GetActor();
651 if (HitActor && HitActor->Implements<UPlayerInteractInterface>())
652 {
653 float Distance = FVector::DistSquared(HitActor->GetActorLocation(), GetActorLocation());
654 if (Distance < tmpDistance)
655 {
656 tmpDistance = Distance;
657 SelectActor = HitActor;
658 }
659 }
660 }
661
662 CurrentInteractTarget = SelectActor;
