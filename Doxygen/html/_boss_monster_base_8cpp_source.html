43void ABossMonsterBase::Die(bool bStageEnded)
44{

46 float CurrentHealth = BossAttributeSet->GetHealth();

48 {
49 bIsDead = true;

54 {

56 }
57 GetCapsuleComponent()->SetCollisionEnabled(ECollisionEnabled::NoCollision);
58

61}

87void ABossMonsterBase::StartDissolveEffect()
88{
89 if (DissolveCurve)
90 {
91 FOnTimelineFloat UpdateFunction;
92 FOnTimelineEvent FinishFunction;
93
94 UpdateFunction.BindUFunction(this, FName("UpdateDissolve"));
95 FinishFunction.BindUFunction(this, FName("OnDissolveFinished"));
96
97 DissolveTimeline.AddInterpFloat(DissolveCurve, UpdateFunction);
98 DissolveTimeline.SetTimelineFinishedFunc(FinishFunction);
99 DissolveTimeline.SetLooping(false);
100 DissolveTimeline.PlayFromStart();
101 }
102}

226void ABossMonsterBase::InitializeAbilitySystem()
227{
228 AbilitySystemComponent->InitAbilityActorInfo(this, this);
229
230 FGameplayEffectContextHandle ContextHandle = AbilitySystemComponent->MakeEffectContext();
231 ContextHandle.AddSourceObject(this);
232 for (TSubclassOf<UGameplayEffect> AttributeEffect : DefaultAttributeEffect)
233 {
234 AbilitySystemComponent->ApplyGameplayEffectToSelf(AttributeEffect->GetDefaultObject<UGameplayEffect>(), 1.f,
235 ContextHandle);
236 }
237
238 for (const TSubclassOf<UGameplayAbility>& Ability : StartUpAbilities)
239 {
240 FGameplayAbilitySpec Spec(Ability, 1, INDEX_NONE);
241 AbilitySystemComponent->GiveAbility(Spec);
242 }
243}
