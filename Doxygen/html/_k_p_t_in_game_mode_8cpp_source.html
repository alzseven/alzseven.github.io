45void AKPTInGameMode::OnPostLogin(AController* NewPlayer)
46{
47 Super::OnPostLogin(NewPlayer);
48
49 // CachedPhaseManager가 아직 유효하지 않으면 아무것도 하지 않고 즉시 리턴합니다.
50 // 이것은 StartPlay가 실행되기 전인 호스트의 첫 로그인 시점에 해당됩니다.
51 if (!IsValid(CachedPhaseManager))
52 {
53 UE_LOG(LogTemp, Log, TEXT("OnPostLogin: PhaseManager not ready yet. This is expected for the host player."));
54 return;
55 }
56
57 // NewPlayer가 유효하고, PlayerController 타입인지 확인합니다.
58 if (APlayerController* PlayerController = Cast<APlayerController>(NewPlayer))
59 {
60 // 이 코드는 나중에 접속하는 클라이언트들을 위해 실행됩니다.
61 // 이 시점에는 CachedPhaseManager가 유효함이 보장됩니다.
62 if (CachedPhaseManager->GetCurrentStateTags().HasTag(FGameplayTag::RequestGameplayTag(TEXT("InGame.Phase.Day"))))
63 {
64 AKPTPlayerState* PlayerState = PlayerController->GetPlayerState<AKPTPlayerState>();
65 if (PlayerState && !PlayerState->GetPersonalTag().IsValid())
66 {
67 UE_LOG(LogTemp, Log, TEXT("Player %s logged in during Day Phase. Initializing..."), *PlayerController->GetName());
68 InitializePlayerForCurrentPhase(PlayerController);
69 }
70 }
71 else
72 {
73 UE_LOG(LogTemp, Log, TEXT("Player %s logged in during Night Phase or Transition. Waiting for next round..."), *PlayerController->GetName());
74 }
75 }
76 else
77 {
78 UE_LOG(LogTemp, Log, TEXT("OnPostLogin: PlayerController is not valid!"));
79 }
80}

119void AKPTInGameMode::StartPlay()
120{
121 Super::StartPlay();
122
123 AKPTInGameState* KPTGameState = GetGameState<AKPTInGameState>();
124 check(KPTGameState && "GameState must be of type AKPTInGameState at InitGame!");
125
126 if (!IsValid(KPTGameState->InGamePhaseManager))
127 {
128 KPTGameState->OnInGamePhaseManagerReady.AddDynamic(this, &AKPTInGameMode::OnInGamePhaseManagerReady);
129 }
130 else
131 {
132 CachedPhaseManager = KPTGameState->InGamePhaseManager;
133 // check(CachedPhaseManager && "InGamePhaseManagerComponent is not valid on GameState!");
134 }
135
136 if (!IsValid(KPTGameState->TeamStatsComponent))
137 {
138 KPTGameState->OnTeamStatsComponentReady.AddDynamic(this, &AKPTInGameMode::OnTeamStatsComponentReady);
139 }
140 else
141 {
142 OnTeamStatsComponentReady(KPTGameState->TeamStatsComponent);
143 // CachedTeamStats = KPTGameState->TeamStatsComponent;
144 // check(CachedTeamStats && "TeamStatsComponent is not valid on GameState!");
145 }
146
147 // 게임에 필요한 핵심 서브시스템들을 미리 캐싱합니다.
148 // Get...() 함수를 호출하여 안전하게 초기화합니다.
149 CachedRewardManager = GetRewardManager();
150 check(CachedRewardManager && "RewardManagerSubsystem could not be initialized!");
151
152 CachedShopManager = GetShopManager();
153 check(CachedShopManager && "ShopManagerSubsystem could not be initialized!");
154
155 CachedStageManager = GetStageManager();
156 check(CachedStageManager && "StageManagerSubsystem could not be initialized!");
157
158 // 보스 관련 델리게이트 바인딩
159 OnBossSpawned.AddDynamic(this, &AKPTInGameMode::HandleBossSpawned);
160 OnBossHPChanged.AddDynamic(this, &AKPTInGameMode::HandleBossHPChanged);
161 OnBossDefeated.AddDynamic(this, &AKPTInGameMode::HandleBossDefeated);
162
163 StartRound();
164}

176void AKPTInGameMode::HandleDayPhase()
177{
178 if (!CachedPhaseManager) return;
179
180 // 1. 전체 게임 상태를 '낮'으로 설정합니다.
181 FGameplayTagContainer DayTags= CachedPhaseManager->GetCurrentStateTags();;
182 DayTags.AddTag(FGameplayTag::RequestGameplayTag(TEXT("InGame.Phase.Day")));
183 CachedPhaseManager->SetStateTags(DayTags);
184
185 // 2. '현재 접속해 있는' 모든 플레이어를 순회하며 초기화합니다.
186 for (FConstPlayerControllerIterator It = GetWorld()->GetPlayerControllerIterator(); It; ++It)
187 {
188 APlayerController* PlayerController = It->Get();
189 InitializePlayerForCurrentPhase(PlayerController); // 분리된 함수 호출
190 }

206 }
207 // Revive if player died on previous round
208 FGameplayTag DeadTag = FGameplayTag::RequestGameplayTag(FName("State.Character.Dead"));

210 {
211 // PlayerController로부터 현재 Pawn을 가져옵니다.

214 // 캐릭터에게 직접 부활 명령을 내립니다.
215 // 부활에 필요한 GE는 GameMode가 건네줄 수 있습니다.

218 }
219

221
222 UPlayerRewardStateComponent* RewardState = PlayerState->GetRewardStateComponent();
223 check(RewardState);
224 //UE_LOG(LogTemp, Type::Display, TEXT("Player %s has %d pending rewards."), *PlayerController->GetName(), RewardState->GetPendingRewardCount());

272void AKPTInGameMode::OnPlayerFinishedRewardChoice(APlayerController* PlayerController)
273{
274 AKPTPlayerState* PlayerState = IsValid(PlayerController)
275 ? PlayerController->GetPlayerState<AKPTPlayerState>()
276 : nullptr;
277 if (!PlayerState) return;
278
279 UPlayerRewardStateComponent* RewardState = PlayerState->GetRewardStateComponent();
280 check(RewardState);
281 // 1. 플레이어의 보상 선택 기회를 하나 소모합니다.
282 RewardState->ConsumePendingReward();
283
284
285 // HandleDayPhase에 있던 로직을 그대로 가져옵니다.
286 if (RewardState->HasPendingRewards())
287 {
288 // 2a. 기회가 남았다면, 다시 보상을 제시합니다. (같은 플레이어에게 연속으로)
289 InitiateRewardPhaseForPlayer(PlayerController);
290 }
291 else
292 {
293 // 2b. 기회를 모두 소모했다면, '상점' 상태로 변경합니다.
294 PlayerState->SetPersonalStateTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::Player_State_InShop));
295 // PlayerIsReadyForNight(PlayerController);
296 // 보상 선택이 모두 끝나면 '상점' 상태로 변경하고 상점 관리자에게 알립니다.
297 // 이 플레이어는 이제 다른 플레이어를 기다릴 필요 없이 상점을 이용할 수 있습니다.
298 if (UShopManagerSubsystem* ShopManager = GetShopManager())
299 {
300 ShopManager->StartShopPhaseForPlayer(PlayerController);
301 }
302 else
303 {
304 UE_LOG(LogTemp, Error, TEXT("InitiateRewardPhaseForPlayer: GetRewardManager() returned nullptr!"));
305 }
306 }

349void AKPTInGameMode::EndNightPhase()
350{
351 GetWorldTimerManager().ClearTimer(NightPhaseTimerHandle);
352 if (CachedPhaseManager)
353 {
354 CachedPhaseManager->SetRemainingTime(0.f);
355 }
356
357 // 패배 조건: 모든 플레이어가 사망했는가?
358 if (AreAllPlayersDead())
359 {
360 HandleMatchDefeat();
361 return;
362 }
363
364 // 승리 조건: 최대 라운드에 도달했는가?
365
366 if (CachedPhaseManager && CachedStageManager && CachedStageManager->IsMaxRound(CachedPhaseManager->GetCurrentRound()))
367 {
368 HandleMatchVictory();
369 return;
370 }
371
372 // 게임이 계속되면 다음 라운드를 준비합니다.
373 if (UStageManagerSubsystem* StageManager = GetStageManager())
374 {
375 StageManager->EndStage(true);
376 }
377 else
378 {
379 UE_LOG(LogTemp, Error, TEXT("InitiateRewardPhaseForPlayer: GetRewardManager() returned nullptr!"));
380 }
381
382 // 밤이 끝난 후, 레벨업을 확인합니다.
383 CheckForTeamLevelUp();
384 ShareTeamGold();
385 InitiatePhaseTransition(FGameplayTag::RequestGameplayTag(TEXT("InGame.Phase.Day")), 2.f);
386}

388void AKPTInGameMode::CheckForTeamLevelUp()
389{
390 if (!CachedTeamStats || !TeamLevelUpTable)
391 {
392 UE_LOG(LogTemp, Error, TEXT("CheckForTeamLevelUp: CachedTeamStats or TeamLevelUpTable is not valid!"));
393 return;
394 }
395
396 // 현재 레벨에 해당하는 데이터 행을 가져옵니다.
397 const FString CurrentLevelRow = FString::FromInt(CachedTeamStats->GetTeamLevel());
398 FTeamLevelUpData* LevelData = TeamLevelUpTable->FindRow<FTeamLevelUpData>(FName(*CurrentLevelRow), TEXT(""));
399 if (!LevelData)
400 {
401 UE_LOG(LogTemp, Error, TEXT("Could not find data for level %d in DT_TeamLevelUp!"), CachedTeamStats->GetTeamLevel());
402 //return;
403 const FString LastLevelRow = FString::FromInt(TeamLevelUpTable->GetRowNames().Num() - 1);
404 LevelData = TeamLevelUpTable->FindRow<FTeamLevelUpData>(FName(*LastLevelRow), TEXT(""));
405 }
406
407 while (CachedTeamStats->GetTeamExperience() >= LevelData->ExperienceForNextLevel)
408 {
409 CachedTeamStats->AddExperience(LevelData->ExperienceForNextLevel * -1);
410 CachedTeamStats->AddLevel(1);
411
412 // 레벨업 보상 풀 타입을 가져옵니다.
413 const ERewardPoolType PoolForThisLevel = LevelData->RewardPool;
414
415 // 레벨업 횟수만큼 모든 플레이어에게 보상 기회와 '컨텍스트'를 부여합니다.
416 for (FConstPlayerControllerIterator It = GetWorld()->GetPlayerControllerIterator(); It; ++It)
417 {
418 if (AController* PC = It->Get())
419 {
420 if (AKPTPlayerState* PS = PC->GetPlayerState<AKPTPlayerState>())
421 {
422 UPlayerRewardStateComponent* RewardState = PS->GetRewardStateComponent();
423 check(RewardState);
424 RewardState->AddPendingReward(PoolForThisLevel);
425 }
426 }
427 }
428
429 // 다음 레벨업에 필요한 경험치량을 위해 새 데이터를 가져옵니다.
430 const FString NewLevelRow = FString::FromInt(CachedTeamStats->GetTeamLevel());
431 LevelData = TeamLevelUpTable->FindRow<FTeamLevelUpData>(FName(*NewLevelRow), TEXT(""));
432 if (!LevelData)
433 {
434 UE_LOG(LogTemp, Warning, TEXT("Max level reached or no data for next level."));
435 //break;
436 const FString LastLevelRow = FString::FromInt(TeamLevelUpTable->GetRowNames().Num() - 1);
437 LevelData = TeamLevelUpTable->FindRow<FTeamLevelUpData>(FName(*LastLevelRow), TEXT(""));
438 }
439 CachedTeamStats->SetExperienceToNextLevel(LevelData->ExperienceForNextLevel);
440 }
441}

459void AKPTInGameMode::CompletePhaseTransition(FGameplayTag TargetPhaseTag)
460{
461 if (CachedPhaseManager)
462 {
463 CachedPhaseManager->SetRemainingTime(0.f);
464 FGameplayTagContainer tempTags = CachedPhaseManager->GetCurrentStateTags();
465 tempTags.RemoveTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::Game_State_Transitioning));
466 //tempTags.AddTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::Game_State_Transitioning));
467 //CachedPhaseManager->SetStateTags(tempTags);
468 //CachedPhaseManager->RemoveStateTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::Game_State_Transitioning));
469
470 if (TargetPhaseTag == FGameplayTag::RequestGameplayTag(GamePhaseTags::InGame_Phase_Night))
471 {
472 //CachedPhaseManager->RemoveStateTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::InGame_Phase_Day));
473 tempTags.RemoveTag(FGameplayTag::RequestGameplayTag(TEXT("InGame.Phase.Day")));
474 CachedPhaseManager->SetStateTags(tempTags);
475 BeginNightPhaseLogic();
476 }
477 else if (TargetPhaseTag == FGameplayTag::RequestGameplayTag(TEXT("InGame.Phase.Day")))
478 {
479 //CachedPhaseManager->RemoveStateTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::InGame_Phase_Night));
480 tempTags.RemoveTag(FGameplayTag::RequestGameplayTag(GamePhaseTags::InGame_Phase_Night));
481 CachedPhaseManager->SetStateTags(tempTags);
482 TeleportPlayerToStart();
483 StartRound();
484 }
485 else if (TargetPhaseTag == FGameplayTag::RequestGameplayTag(GamePhaseTags::InGame_Phase_GameOver))
486 {
487 HandleMatchHasEnded();
488 }
489 else
490 {
491 UE_LOG(LogTemp, Warning, TEXT("CompletePhaseTransition: Unknown phase tag '%s'"), *TargetPhaseTag.ToString());
492 }
493 }
