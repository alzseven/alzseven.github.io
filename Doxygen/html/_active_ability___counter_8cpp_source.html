52void UActiveAbility_Counter::ApplyEffectToTarget(AActor* Target, float DamageAmount)
53{
54 if (!PayloadEffectClass || !Target) return;
55
56 UAbilitySystemComponent* SourceASC = GetAbilitySystemComponentFromActorInfo();
57 UAbilitySystemComponent* TargetASC = UAbilitySystemBlueprintLibrary::GetAbilitySystemComponent(Target);
58 if (!SourceASC || !TargetASC) return;
59
60 FGameplayEffectContextHandle EffectContext = SourceASC->MakeEffectContext();
61 EffectContext.AddSourceObject(SourceASC->GetOwner());
62 FGameplayEffectSpecHandle SpecHandle = SourceASC->MakeOutgoingSpec(PayloadEffectClass, GetAbilityLevel(), EffectContext);
63 if (SpecHandle.IsValid())
64 {
65 float FinalDamage = BaseDamage;
66 if (TargetASC->HasMatchingGameplayTag(FGameplayTag::RequestGameplayTag(FName("Enemy.State.Groggy"))))
67 {
68 FinalDamage *= 10.f;
69 }
70 UAbilitySystemBlueprintLibrary::AssignTagSetByCallerMagnitude(SpecHandle, DamageTag, FinalDamage);
71 UAbilitySystemBlueprintLibrary::AssignTagSetByCallerMagnitude(SpecHandle, CriticalTag, 1.f);
72 UAbilitySystemBlueprintLibrary::AssignTagSetByCallerMagnitude(SpecHandle, CriticalMultiplierTag, 5.f);
73
74 TargetASC->ApplyGameplayEffectSpecToSelf(*SpecHandle.Data.Get());
75 }
76}

78void UActiveAbility_Counter::SendCounterEventToTarget(AActor* Target)
79{
80 if (!Target) return;
81 UAbilitySystemComponent* TargetASC = UAbilitySystemBlueprintLibrary::GetAbilitySystemComponent(Target);
82 if (TargetASC && TargetASC->HasMatchingGameplayTag(FGameplayTag::RequestGameplayTag(FName("Enemy.State.CanCounter"))))
83 {
84 FGameplayTag CounterEventTag = FGameplayTag::RequestGameplayTag(FName("Event.Counter.Hit"));
85 FGameplayEventData PayloadEvent;
86 PayloadEvent.Instigator = GetAvatarActorFromActorInfo();
87 PayloadEvent.Target = Target;
88 UAbilitySystemBlueprintLibrary::SendGameplayEventToActor(Target, CounterEventTag, PayloadEvent);
89 TargetASC->RemoveLooseGameplayTag(FGameplayTag::RequestGameplayTag(FName("Enemy.State.CanCounter")));
90 }
91}
