10bool USkillBuffSystemStatics::ApplyBuff(UAbilitySystemComponent* TargetASC, FName TargetSkillTag, FName BuffID, ECardRarity Rarity)
11{
12 // ASC 불러오기
13 if (!IsValid(TargetASC))
14 {
15 UE_LOG(LogTemp, Error, TEXT("ApplyBuff Failed: TargetASC is invalid."));
16 return false;
17 }
18 FGameplayTag SkillTag = FGameplayTag::RequestGameplayTag(TargetSkillTag,false);
19 static const FString BuffTablePath = TEXT("DataTable'/Game/Blueprints/Skill/Data/DT_SkillBuffTable.DT_SkillBuffTable'");
20 const UDataTable* BuffTable = Cast<UDataTable>(StaticLoadObject(UDataTable::StaticClass(), nullptr, *BuffTablePath));
21
22 // 데이터 테이블 확인
23 if (!BuffTable)
24 {
25 UE_LOG(LogTemp, Error, TEXT("ApplyBuff Failed: Cannot load DataTable at %s."), *BuffTablePath);
26 return false;
27 }
28
29 // 데이터 테이블 행 확인
30 const FSkillBuffData* BuffData = BuffTable->FindRow<FSkillBuffData>(BuffID, TEXT(""));
31 if (!BuffData)
32 {
33 UE_LOG(LogTemp, Warning, TEXT("ApplyBuff Failed: Cannot find BuffID '%s' in DT_Buffs."), *BuffID.ToString());
34 return false;
35 }
36
37 // Base GE 존재 확인
38 if (!BuffData->BaseGameplayEffect)
39 {
40 UE_LOG(LogTemp, Warning, TEXT("ApplyBuff Failed: BaseGameplayEffect is not set for BuffID '%s'."), *BuffID.ToString());
41 return false;
42 }
43
44 // GE 스펙핸들 생성 확인
45 FGameplayEffectSpecHandle SpecHandle = TargetASC->MakeOutgoingSpec(BuffData->BaseGameplayEffect, 1.0f, TargetASC->MakeEffectContext());
46 if (!SpecHandle.IsValid())
47 {
48 return false;
49 }
50
51 // Duration 데이터가 -1이면 Infinite로 처리됨
52 SpecHandle.Data->SetDuration(BuffData->Duration, true);
53
54 // FGameplayTag UniqueBuffTag = FGameplayTag::RequestGameplayTag(BuffID, false);
55 // if(UniqueBuffTag.IsValid())
56 // {
57 // SpecHandle.Data->AddDynamicAssetTag(UniqueBuffTag);
58 // }
59 //SpecHandle.Data->DynamicGrantedTags.AppendTags(BuffData->GrantedBuffTags);
60 SpecHandle.Data->AppendDynamicAssetTags(BuffData->GrantedBuffTags);
61
62 if (SkillTag.IsValid())
63 {
64 //SpecHandle.Data->DynamicGrantedTags.AddTag(TargetSkillTag);
65 SpecHandle.Data->AddDynamicAssetTag(SkillTag);
66 }
67 // Modifiers를 순회하며 등급에 맞는 수치를 적용
68 for (const FBuffModifierData& ModData : BuffData->Modifiers)
69 {
70 // 1. 등급에 맞는 수치를 가져옵니다.
71 const float FinalMagnitude = ModData.Magnitude.GetMagnitudeForRarity(Rarity);
72
73 // 2. 해당 수치를 SetByCaller로 설정합니다.
74 SpecHandle.Data->SetSetByCallerMagnitude(ModData.DataTag, FinalMagnitude);
75 }
76
77 TargetASC->ApplyGameplayEffectSpecToSelf(*SpecHandle.Data.Get());
78
79 return true;
80}

82FDetailedSkillData USkillBuffSystemStatics::GetSkillStatus(UAbilitySystemComponent* TargetASC, FName TargetSkillTag)
83{
84 // ASC 불러오기
85 if (!IsValid(TargetASC))
86 {
87 UE_LOG(LogTemp, Error, TEXT("ApplyBuff Failed: TargetASC is invalid."));
88 return FDetailedSkillData();
89 }
90 FGameplayTag SkillTag = FGameplayTag::RequestGameplayTag(TargetSkillTag,false);
91 TArray<FGameplayAbilitySpec>& FindSpecs = TargetASC->GetActivatableAbilities();
92 UBaseSkillAbility* FindAbility = nullptr;
93 for (FGameplayAbilitySpec& Spec : FindSpecs)
94 {
95 if (Spec.Ability && Spec.Ability->AbilityTags.HasTag(SkillTag))
96 {
97 FindAbility = Cast<UBaseSkillAbility>(Spec.GetAbilityInstances()[0]);
98 break;
99 }
100 }
101 if (!FindAbility)
102 {
103 return FDetailedSkillData();
104 }
105
106 return FindAbility->CalculatedStatusData;
107}
