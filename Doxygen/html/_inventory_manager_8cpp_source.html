82bool UInventoryManager::AddItem(const FName& InItemName)
83{
84 // 서순 바꿈 - 준형 250703
85 if (!ItemDataTable)
86 {
87 return false;
88 }
89
90 if (IsInventoryFull())
91 {
92 return false;
93 }
94
95 // TODO ItemDataTable 가져오기
96 FItemDataRow* ItemDef = ItemDataTable->FindRow<FItemDataRow>(InItemName, TEXT(""));
97
98 check(ItemDef);
99
100 FItemInstance ItemInstance = FItemInstance::CreateItemByDef(ItemDef);
101
102 IAbilitySystemInterface* AbilityOwner = Cast<IAbilitySystemInterface>(GetOwner());
103
104 check(AbilityOwner);
105
106 UAbilitySystemComponent* ASC = AbilityOwner ? AbilityOwner->GetAbilitySystemComponent() : nullptr;
107
108 check(ASC);
109
110 switch (ItemDef->ItemType)
111 {
112 case EKPTItemType::StatOnly:
113 {
114 // GE 적용 패턴
115 FGameplayEffectSpecHandle SpecHandle = ASC->MakeOutgoingSpec(ItemDef->ItemBuffEffect, 1.f, ASC->MakeEffectContext());
116 if (SpecHandle.IsValid())
117 {
118 const FGameplayTag& EffectTag = ItemDef->AttributeTag;
119 float Magnitude = ItemDef->Multiplier;
120
121 // 3. SetByCallerMagnitude 설정
122 SpecHandle.Data->SetSetByCallerMagnitude(EffectTag, Magnitude);
123
124 ItemInstance.GrantedEffectHandle = ASC->ApplyGameplayEffectSpecToSelf(*SpecHandle.Data);
125 }
126 }
127 break;
128
129 case EKPTItemType::PassiveAbility:
130 {
131 //Item이 가진 Ability를 ASC에 부여하고, 어떤 이벤트에 ActivateAbility가 호출될지 정해주어야한다.
132 if (ItemDef->PassiveAbilityClass)
133 {
134 FGameplayAbilitySpec AbilitySpec(ItemDef->PassiveAbilityClass, 1, -1, this);
135
136 ItemInstance.GrantedAbilityHandle = ASC->GiveAbility(AbilitySpec);
137 if (ItemInstance.GrantedAbilityHandle.IsValid() == false)
138 {
139 UE_LOG(LogTemp, Error, TEXT("ItemAbility Give Failed : InventoryManager"));
140 }
141 }
142 }
143 break;
144 }
145
146 EquippedItems.Add(ItemInstance);
147
148
149 EquippedUIItems.Add(ItemDef->ItemId);
150
151
152 // MARK_PROPERTY_DIRTY_FROM_NAME(UInventoryManager, EquippedItems, this);
153 OnInventoryUpdated.Broadcast(EquippedItems);
154 OnRep_EquippedItems();
155
156 OnInventoryChanged.Broadcast(EquippedUIItems);
157 OnRep_EquippedUIItems();
158
159
160 DisplayInventory();
161 return true;
162}

189bool UInventoryManager::RemoveItem(int32 SlotIndex)
190{
191 if (!EquippedItems.IsValidIndex(SlotIndex))
192 return false;
193
194 // 1. 아이템 인스턴스 꺼내기
195 FItemInstance& ItemInstance = EquippedItems[SlotIndex];
196
197 // 1. Owner에서 ASC 인터페이스 찾기
198 IAbilitySystemInterface* AbilityOwner = Cast<IAbilitySystemInterface>(GetOwner());
199 if (AbilityOwner && ItemInstance.GrantedAbilityHandle.IsValid())
200 {
201 // 2. ASC 가져오기
202 UAbilitySystemComponent* ASC = AbilityOwner->GetAbilitySystemComponent();
203 if (ASC)
204 {
205 FGameplayAbilitySpec* AbilitySpec = ASC->FindAbilitySpecFromHandle(ItemInstance.GrantedAbilityHandle);
206
207 UItemAbility* ItemAbility = Cast<UItemAbility>(AbilitySpec->Ability);
208 // 3. 부여한 Ability 제거
209 ASC->ClearAbility(ItemInstance.GrantedAbilityHandle);
210
211 // 4. Handle 초기화
212 ItemInstance.GrantedAbilityHandle = FGameplayAbilitySpecHandle();
213 }
214 }
215
216 if (ItemInstance.GrantedEffectHandle.IsValid())
217 {
218 // 2. ASC 가져오기
219 UAbilitySystemComponent* ASC = AbilityOwner->GetAbilitySystemComponent();
220 if (ASC)
221 {
222 ASC->RemoveActiveGameplayEffect(ItemInstance.GrantedEffectHandle);
223 ItemInstance.GrantedEffectHandle.Invalidate(); // 내 핸들 변수 초기화 (재사용/중복 방지)
224 }
225 }
226
227 // 3. 슬롯에서 제거
228 EquippedItems.RemoveAt(SlotIndex);
229 OnRep_EquippedItems();
230
231 EquippedUIItems.RemoveAt(SlotIndex);
232 OnRep_EquippedUIItems();
233
234 return true;
235}
