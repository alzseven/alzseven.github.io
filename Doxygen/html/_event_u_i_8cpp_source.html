112void UEventUI::StartGrayscaleFade(bool bFadeToGrayscale, float Duration)
113{
114 UWorld* World = GetWorld();
115
116 if (!World)
117 {
118 return;
119 }
120
121
122 if (!TargetPostProcessVolume)
123 {
124 for (TActorIterator<APostProcessVolume> It(World); It; ++It)
125 {
126 // Infinite Extent(Unbound)가 활성화된 볼륨을 찾습니다.
127 if ((*It)->bUnbound)
128 {
129 TargetPostProcessVolume = *It;
130 break; // 첫 번째로 찾은 것을 사용
131 }
132 }
133 }
134
135 if (!TargetPostProcessVolume)
136 {
137 return;
138 }
139
140 if (bFadeToGrayscale)
141 {
142 // 흑백으로 전환: 1 (원래 색상) -> 0 (흑백)
143 StartSaturation = 1.0f;
144 EndSaturation = 0.0f;
145 }
146 else
147 {
148 // 원래 색상으로 복원: 0 (흑백) -> 1 (원래 색상)
149 StartSaturation = 0.0f;
150 EndSaturation = 1.0f;
151 }
152
153 // 페이드 지속 시간과 시작 시간을 기록합니다.
154 FadeDuration = Duration;
155 FadeStartTime = World->GetTimeSeconds();
156
157 // 채도 값을 덮어쓸 것이라고 명시합니다.
158 TargetPostProcessVolume->Settings.bOverride_ColorSaturation = true;
159
160 // 타이머를 설정하여 UpdateGrayscaleFade 함수를 매 프레임 호출하도록 합니다.
161 // 0.016초는 약 60fps에 해당하며, 더 부드러운 전환을 위해 짧은 간격으로 설정합니다.
162 World->GetTimerManager().SetTimer(
163 GrayscaleFadeTimerHandle,
164 this,
165 &UEventUI::UpdateGrayscaleFade,
166 0.016f,
167 true);
168}

170void UEventUI::UpdateGrayscaleFade()
171{
172 // 유효한 월드와 포스트 프로세스 볼륨이 있는지 다시 확인합니다.
173 UWorld* World = GetWorld();
174 if (!World || !TargetPostProcessVolume)
175 {
176 // 타이머를 안전하게 정지합니다.
177 World->GetTimerManager().ClearTimer(GrayscaleFadeTimerHandle);
178 return;
179 }
180
181 // 페이드 시작 후 경과된 시간을 계산합니다.
182 const float ElapsedTime = World->GetTimeSeconds() - FadeStartTime;
183
184 // 진행률(Alpha)을 계산합니다. (0.0 ~ 1.0 사이 값)
185 const float Alpha = FMath::Clamp(ElapsedTime / FadeDuration, 0.0f, 1.0f);
186
187 // Lerp (선형 보간) 함수를 사용해 시작과 끝 채도 사이의 현재 값을 계산합니다.
188 const float CurrentSaturation = FMath::Lerp(StartSaturation, EndSaturation, Alpha);
189
190 // 계산된 채도 값을 포스트 프로세스 설정에 적용합니다.
191 TargetPostProcessVolume->Settings.ColorSaturation.W = CurrentSaturation;
192
193 // 페이드가 완료되었는지 확인합니다. (Alpha가 1.0에 도달)
194 if (Alpha >= 1.0f)
195 {
196 // 타이머를 정지시켜 더 이상 업데이트하지 않도록 합니다.
197 World->GetTimerManager().ClearTimer(GrayscaleFadeTimerHandle);
198
199 // 만약 색상을 복원한 것이라면, 굳이 값을 덮어쓸 필요가 없으므로 Override 설정을 해제할 수 있습니다.
200 if (EndSaturation == 1.0f)
201 {
202 TargetPostProcessVolume->Settings.bOverride_ColorSaturation = false;
203 }
204 }
