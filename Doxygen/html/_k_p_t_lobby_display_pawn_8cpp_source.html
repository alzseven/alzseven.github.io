38void AKPTLobbyDisplayPawn::AttemptInitialStateUpdate()
39{
40 if (CachedLobbyGameState)
41 {
42 // OnLobbyStateUpdated를 호출하여 bIsDisplayingValidInfo 플래그를 갱신하도록 합니다.
43 OnLobbyStateUpdated(CachedLobbyGameState->PlayerSlots);
44
45 // 업데이트 함수가 플래그를 true로 설정했다면 타이머를 중지합니다.
46 if (bIsDisplayingValidInfo)
47 {
48 GetWorldTimerManager().ClearTimer(InitialStateUpdateTimer);
49 }
50 }
51}

54void AKPTLobbyDisplayPawn::OnLobbyStateUpdated(const TArray<FPlayerSlotInfo>& NewPlayerSlots)
55{
56 if (!CachedLobbyGameState) return;
57
58 // ... (자리 바꾸기 로직은 동일) ...
59 const APlayerController* LocalPC = UGameplayStatics::GetPlayerController(this, 0);
60 const APlayerState* LocalPlayerState = LocalPC ? LocalPC->PlayerState : nullptr;
61 FGameplayTag LocalPlayerSlotTag;
62
63 if (LocalPlayerState)
64 {
65 for (const FPlayerSlotInfo& Slot : CachedLobbyGameState->PlayerSlots)
66 {
67 if (Slot.PlayerState == LocalPlayerState)
68 {
69 LocalPlayerSlotTag = Slot.SlotTag;
70 break;
71 }
72 }
73 }
74
75 FGameplayTag TargetSlotTagToShow = SlotIdentifierTag;
76
77 const FGameplayTag CenterSlotTag = FGameplayTag::RequestGameplayTag("Lobby.PlayerStart.Center");
78
79 if (SlotIdentifierTag == CenterSlotTag)
80 {
81 TargetSlotTagToShow = LocalPlayerSlotTag;
82 }
83 else if (LocalPlayerSlotTag == CenterSlotTag)
84 {
85 TargetSlotTagToShow = SlotIdentifierTag;
86 }
87 else if (LocalPlayerSlotTag == SlotIdentifierTag)
88 {
89 TargetSlotTagToShow = CenterSlotTag;
90 }
91
92
93 // 로컬 플레이어가 아직 입장하지 않아 슬롯 태그가 없을 수 있습니다.
94 // 이 경우 중앙 마네킹은 아직 누구를 표시해야 할지 모르므로 업데이트는 '실패'한 것입니다.
95 if (SlotIdentifierTag == CenterSlotTag && !LocalPlayerSlotTag.IsValid())
96 {
97 bIsDisplayingValidInfo = false;
98 UpdateDisplay(FPlayerSlotInfo()); // 빈 정보로 업데이트
99 return;
100 }
101
102 for (const FPlayerSlotInfo& Slot : CachedLobbyGameState->PlayerSlots)
103 {
104 if (Slot.SlotTag == TargetSlotTagToShow)
105 {
106 UpdateDisplay(Slot);
107 // 유효한 슬롯 정보를 찾았으므로 성공 플래그를 설정합니다.
108 bIsDisplayingValidInfo = true;
109 return;
110 }
111 }
112
113 // 해당하는 슬롯을 아직 찾지 못했다면 비어있는 상태로 둡니다.
114 UpdateDisplay(FPlayerSlotInfo());
115 bIsDisplayingValidInfo = false; // 아직 성공 아님
116}
