116void UShopUI::NativeConstruct()
117{
118 Super::NativeConstruct();
119
120 UE_LOG(LogTemp, Warning, TEXT("ShopUI Create!"));
121 if (Close)
122 {
123 Close->OnClicked.RemoveDynamic(this, &UShopUI::HandleClickedClose);
124 Close->OnClicked.AddDynamic(this, &UShopUI::HandleClickedClose);
125 }
126
127 if (RerollButton)
128 {
129 RerollButton->OnClicked.RemoveDynamic(this, &UShopUI::HandleClickedReroll);
130 RerollButton->OnClicked.AddDynamic(this, &UShopUI::HandleClickedReroll);
131 }
132
133 ShopGrid->ClearChildren();
134
135 for (int i = 0; i < MaxItemCount; ++i)
136 {
137 if (ItemSlotUIClass)
138 {
139 Shop.Add(CreateWidget<UItemSlotUI>(GetWorld(), ItemSlotUIClass));
140 Shop[i]->SetTooltipClass(ItemTooltipUIClass);
141 ShopGrid->AddChildToUniformGrid(Shop[i],i / MaxGridRow, i % MaxGridRow);
142 }
143
144 else
145 {
146 break;
147 }
148 }
149
150 if (AKPTPlayerState* PlayerState = Cast<AKPTPlayerState>(GetOwningPlayerState()))
151 {
152 OnPlayerStateSet();
153 }
154}
