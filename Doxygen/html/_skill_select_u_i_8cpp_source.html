12void USkillSelectUI::NativeConstruct()
13{
14 Super::NativeConstruct();
15
16 if (AKPTPlayerState* PlayerState = Cast<AKPTPlayerState>(GetOwningPlayerState()))
17 {
18 if (UPlayerRewardStateComponent* RewardState = PlayerState->GetRewardStateComponent())
19 {
20 RewardState->OnRewardChoicesChanged.RemoveDynamic(this, &USkillSelectUI::UpdateSkillSelectSlot);
21 RewardState->OnRewardChoicesChanged.AddDynamic(this, &USkillSelectUI::UpdateSkillSelectSlot);
22 }
23
24 if(PlayerSkillSlot0 && PlayerSkillSlot1 && PlayerSkillSlot2 && PlayerSkillSlot3)
25 {
26 PlayerEquipSkillSlots.Add(PlayerSkillSlot0);
27 PlayerEquipSkillSlots.Add(PlayerSkillSlot1);

29 PlayerEquipSkillSlots.Add(PlayerSkillSlot3);
30 }
31
32 if (LeftSkill && CenterSkill && RightSkill)
33 {
34 LeftSkill->OnHoveredSkillSlotUI.RemoveDynamic(this, &USkillSelectUI::UpdateHoverdSkillsVisual);
35 LeftSkill->OnHoveredSkillSlotUI.AddDynamic(this, &USkillSelectUI::UpdateHoverdSkillsVisual);

41 LeftSkill->OnUnHoveredSkillSlotUI.RemoveDynamic(this, &USkillSelectUI::UpdateUnHoverdSkillsVisual);

46 RightSkill->OnUnHoveredSkillSlotUI.AddDynamic(this, &USkillSelectUI::UpdateUnHoverdSkillsVisual);

48
49 if (APlayerCharacter* Player= Cast<APlayerCharacter>(GetOwningPlayerPawn()))

53 SkillManager->OnAddSkill.RemoveDynamic(this, &USkillSelectUI::UpdateSkillSlotData);

143void USkillSelectUI::UpdateSkillSlotData(const TArray<FName>& SkillNames)
144{
145 if (SkillDataTable)
146 {
147 for (int i = 0; i < SkillNames.Num(); ++i)
148 {
149 if (i >= PlayerEquipSkillSlots.Num())
150 {
151 break;
152 }
153
154 FSkillData* SkillData = SkillDataTable->FindRow<FSkillData>(SkillNames[i], "");
155
156 if (SkillData)
157 {
158 PlayerEquipSkillSlots[i]->UpdateSlotVisual(*SkillData);
159 }
160 }

164void USkillSelectUI::UpdateHoverdSkillsVisual(USkillSlotUI* SelectSlot)
165{
166 FName RowName = SelectSlot->GetCachedSkillId();
167
168 FSkillData* SkillData = SkillDataTable->FindRow<FSkillData>(RowName, "");
169
170 if (SkillData)
171 {
172 for (int i = 0; i < PlayerEquipSkillSlots.Num(); ++i)
173 {
174 PlayerEquipSkillSlots[i]->SelectMatchingVisual(false);
175 }
176
177 //bool Is
178 for (int i = 0; i < PlayerEquipSkillSlots.Num(); ++i)
179 {
180 FName Temp = SkillData->SkillName;
181 FName Dest = PlayerEquipSkillSlots[i]->GetCachedSkillData().SkillName ;
182
183 if (Temp == Dest)
184 {
185 PlayerEquipSkillSlots[i]->SelectMatchingVisual(true);
186 break;
187 }
188
189 else if(Dest.IsNone())
190 {
191 PlayerEquipSkillSlots[i]->SelectMatchingVisual(true);
192 PlayerEquipSkillSlots[i]->PreUpdateSlotVisual(*SkillData);
193 break;
194 }
195 }
196 }
