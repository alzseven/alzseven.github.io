91void UFlameEmblemAbility::ActivateAbility(const FGameplayAbilitySpecHandle Handle,
92 const FGameplayAbilityActorInfo* ActorInfo, const FGameplayAbilityActivationInfo ActivationInfo,
93 const FGameplayEventData* TriggerEventData)
94{
95 Super::ActivateAbility(Handle, ActorInfo, ActivationInfo, TriggerEventData);
96
97 UAbilitySystemComponent* ASC = ActorInfo->AbilitySystemComponent.Get();
98
99 const AActor* TargetActor = TriggerEventData->Target;
100 if (TargetActor == nullptr)
101 {
102 EndAbility(Handle, ActorInfo, ActivationInfo, true, false);
103 return;
104 }
105
106 const IAbilitySystemInterface* TargetASI = Cast<IAbilitySystemInterface>(TargetActor);
107
108 if (TargetASI == nullptr)
109 {
110 EndAbility(Handle, ActorInfo, ActivationInfo, true, false);
111 return;
112 }
113
114 UAbilitySystemComponent* TargetASC = TargetASI->GetAbilitySystemComponent();
115 if (TargetASC == nullptr)
116 {
117 EndAbility(Handle, ActorInfo, ActivationInfo, true, false);
118 return;
119 }
120
121 // 자신이 자신한테 부여하는 경우는 종료
122 if (ASC == TargetASC)
123 {
124 EndAbility(Handle, ActorInfo, ActivationInfo, true, false);
125 return;
126 }
127
128 const FGameplayEffectContextHandle& EffectContextHandle = ASC->MakeEffectContext();
129 FGameplayEffectSpecHandle EffectSpecHandle = ASC->MakeOutgoingSpec(FlameEffect, 1.f, EffectContextHandle);
130 TargetASC->ApplyGameplayEffectSpecToSelf(*EffectSpecHandle.Data.Get());
131
132 EndAbility(Handle, ActorInfo, ActivationInfo, true, false);
133}
