40void ALandingIndicator::OnOverlapBegin(UPrimitiveComponent* OverlappedComp, AActor* OtherActor,
41 UPrimitiveComponent* OtherComp, int32 OtherBodyIndex, bool bFromSweep, const FHitResult& SweepResult)
42{
43 if (HasAuthority())
44 {
45 ACharacter* PlayerCharacter = Cast<ACharacter>(OtherActor);
46 if (PlayerCharacter && PlayerCharacter->IsPlayerControlled())
47 {
48 PlayerCount++;
49 return; // 플레이어만 감지하고 종료
50 }
51
52 ABouncingBall* Ball = Cast<ABouncingBall>(OtherActor);
53 if (Ball && MissionOwner)
54 {
55 // 공이 들어왔음을 미션 관리자에게 알림
56 MissionOwner->OnBallEnteredIndicator(Ball);
57 }
58 }
59}

61void ALandingIndicator::OnOverlapEnd(UPrimitiveComponent* OverlappedComp, AActor* OtherActor,
62 UPrimitiveComponent* OtherComp, int32 OtherBodyIndex)
63{
64 ACharacter* PlayerCharacter = Cast<ACharacter>(OtherActor);
65 if (PlayerCharacter && PlayerCharacter->IsPlayerControlled())
66 {
67 PlayerCount--;
68 }
69}
