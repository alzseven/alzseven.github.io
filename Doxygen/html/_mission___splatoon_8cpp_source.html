16void AMission_Splatoon::StartMission()
17{
18 Super::StartMission();
19 if (!HasAuthority()) return;
20
21 if (PaintableVolumeClass)
22 {
23 // 이 관리자 액터의 위치에 구성요소(PaintableVolume)를 스폰합니다.
24 // 이 위치는 MissionManager가 TargetPoint를 통해 지정해준 위치입니다.
25 FActorSpawnParameters SpawnParams;
26 SpawnParams.Owner = this;
27 SpawnedVolume = GetWorld()->SpawnActor<APaintableVolume>(PaintableVolumeClass, GetActorLocation(), FRotator::ZeroRotator, SpawnParams);
28
29 if (SpawnedVolume)
30 {
31 // 구성요소가 상태를 업데이트할 때마다 OnPaintableVolumeStateUpdated 함수를 호출하도록 바인딩합니다.
32 SpawnedVolume->OnStateUpdated.AddDynamic(this, &AMission_Splatoon::OnPaintableVolumeStateUpdated);
33 }
34 }
35
36 UE_LOG(LogTemp, Log, TEXT("Paint The Area Mission Manager Initialized."));
37}

39void AMission_Splatoon::OnPaintableVolumeStateUpdated()
40{
41 // [수정] 부모의 bMissionHandled 플래그 덕분에, 이 함수가 호출되더라도
42 // 이미 완료/실패 처리된 후라면 아무것도 하지 않습니다.
43 // 하지만 만약을 위해 한 번 더 체크합니다.
44 if (!HasAuthority() || !SpawnedVolume) return;
45
46 // MissionActorBase의 bMissionHandled가 true이면 더 이상 진행하지 않도록 할 수 있습니다.
47 // 하지만 지금은 Notify... 함수 내부의 가드가 충분합니다.
48
49 const int32 PaintedCells = SpawnedVolume->GetRemainingPaintedCells(); // 남은 오염 그리드 개수
50 const int32 TotalCells = SpawnedVolume->GetTotalValidGridCells(); // 전체 오염 그리드 개수
51
52 if (TotalCells > 0)
53 {
54 MissionProgress = 1.0f - ((float)PaintedCells / (float)TotalCells);
55 }
56
57 // 완료 조건을 확인합니다.
58 if (MissionProgress >= GetCompletionThreshold())
59 {
60 // [핵심 수정] 부모의 안전한 함수를 호출하고, 그 이후에 아무것도 하지 않습니다.
61 Super::NotifyMissionComplete();
62 }
63}
