41
42 Damage->SetText(FText::AsNumber(Amount));
43 bIsCritical = IsCritical;
44
45 UWidgetAnimation* CurrentAnimation = nullptr;
46 if (bIsCritical)

48 CurrentAnimation = CriticalDamageAnimation;

51 {
