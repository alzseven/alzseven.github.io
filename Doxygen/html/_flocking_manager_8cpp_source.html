357FVector2D AFlockingManager::CalculatePathfind(int32 EnemyIndex)
358{
359 UEnemyFlockingComponent* Enemy = Enemies[EnemyIndex];
360
361 check(Enemy);
362
363 if (Enemy->bPathfinding == false)
364 return FVector2D::ZeroVector;
365
366 if (!Enemy->PathPoints.IsValidIndex(Enemy->CurrentPathIndex))
367 {
368 Enemy->bPathfinding = false;
369 return FVector2D::ZeroVector;
370 }
371
372 //해당 PathPoints로 이동하기
373 FVector2D CurrentPos = FVector2D(FlockDataCache[EnemyIndex].Position);
374 FVector2D TargetPos = FVector2D(Enemy->PathPoints[Enemy->CurrentPathIndex].X, Enemy->PathPoints[Enemy->CurrentPathIndex].Y);
375
376 FVector2D ToTarget = TargetPos - CurrentPos;
377
378 // 여기서 LastDir 정의 (현재 도착하려는 지점 - 이전 지점)
379 FVector2D LastDir = FVector2D::ZeroVector;
380
381 if (Enemy->CurrentPathIndex > 0)
382 {
383 FVector2D PrevTarget = FVector2D(Enemy->PathPoints[Enemy->CurrentPathIndex - 1]);
384 LastDir = (TargetPos - PrevTarget).GetSafeNormal();
385 }
386 else
387 {
388 LastDir = ToTarget.GetSafeNormal(); // 시작점에서는 이걸로
389 }
390
391 if (FVector2D::DotProduct(LastDir, ToTarget) < 0)
392 {
393 Enemy->CurrentPathIndex++;
394
395 // 다음 목적지도 없으면 종료
396 if (!Enemy->PathPoints.IsValidIndex(Enemy->CurrentPathIndex))
397 {
398 Enemy->bPathfinding = false;
399 return FVector2D::ZeroVector;
400 }
401 }
402
403 const FVector2D NewTarget = FVector2D(Enemy->PathPoints[Enemy->CurrentPathIndex]);
404
405 return (NewTarget - CurrentPos).GetSafeNormal();
406}

408FVector2D AFlockingManager::CalculateSeparation(int32 EnemyIndex)
409{
410 FVector2D SteerForce = FVector2D::ZeroVector;
411 int32 Count = 0;
412
413 const FFlockingData& CurrentData = FlockDataCache[EnemyIndex];
414 float SeparationRadiusSq = FMath::Square(Enemies[EnemyIndex]->SeparationRadius);
415
416 // 공간 해시를 사용한 효율적인 이웃 검색
417 TArray<int32> PotentialNeighbors = SpatialGrid.GetNearbyEntities(
418 CurrentData.Position,
419 Enemies[EnemyIndex]->SeparationRadius
420 );
421
422 bool bCurrentPathfinding = Enemies[EnemyIndex]->bPathfinding;
423
424 for (int32 NeighborIdx : PotentialNeighbors)
425 {
426 if (NeighborIdx == EnemyIndex) continue;
427
428 //현재 유닛이 길찾기 중이라면
429 //다른 유닛과는 분리를 하면 안된다.
430 //길찾기 유닛끼리만 분리해야한다.
431 if (bCurrentPathfinding && !Enemies[NeighborIdx]->bPathfinding)
432 continue;
433 //다른 유닛이 사망 상태라면 무시
434 if (Enemies[NeighborIdx]->bCanMove == false)
435 continue;
436
437 const FFlockingData& NeighborData = FlockDataCache[NeighborIdx];
438
439 //TODO 우선순위가 있다면?
440 FVector2D Diff = CurrentData.Position - NeighborData.Position;
441 float DistanceSq = Diff.SizeSquared();
442
443 if (DistanceSq > 0 && DistanceSq < SeparationRadiusSq)
444 {
445 // 거리에 반비례하는 힘
446 float Distance = FMath::Sqrt(DistanceSq);
447 Diff.Normalize();
448 Diff /= Distance; // 가까울수록 강한 힘
449 SteerForce += Diff;
450 Count++;
451 }
452 }
453
454 if (Count > 0)
455 {
456 SteerForce /= static_cast<float>(Count);
457
458 // Reynolds의 조향 공식
459 if (SteerForce.SizeSquared() > 0)
460 {
461 SteerForce.Normalize();
462 SteerForce *= Enemies[EnemyIndex]->MaxSpeed;
463 SteerForce -= CurrentData.Velocity;
464 }
465 }
466
467 return SteerForce;
468}

470FVector2D AFlockingManager::CalculateTargetSeek(int32 EnemyIndex)
471{
472 AActor* TargetActor = Enemies[EnemyIndex]->Target;
473 if (TargetActor == nullptr)
474 {
475 UE_LOG(LogTemp, Error, TEXT("FlockingManager: Target이 없습니다."));
476 return FVector2D::ZeroVector;
477 }
478 const FFlockingData& CurrentData = FlockDataCache[EnemyIndex];
479
480 FVector2D TargetLocation = FVector2D(TargetActor->GetActorLocation().X, TargetActor->GetActorLocation().Y);
481
482 FVector2D ToTarget = (TargetLocation - CurrentData.Position).GetSafeNormal();
483
484 FVector2D DesiredVel = ToTarget * Enemies[EnemyIndex]->MaxSpeed;
485
486 // 5) 스티어링(Force) = Desired Velocity − 현재 Velocity
487 //FVector2D Steering = DesiredVel - CurrentData.Velocity;
488 FVector2D Steering = DesiredVel;
489
490 Enemies[EnemyIndex]->CachedToTarget = Steering;
491
492 return Steering;
493}

495FVector2D AFlockingManager::CalculateObstacleAvoidance(int32 EnemyIndex, FVector ForceDir)
496{
497 FVector2D AvoidForce = FVector2D::ZeroVector;
498 const FFlockingData& CurrentData = FlockDataCache[EnemyIndex];
499 AActor* TargetOwner = Enemies[EnemyIndex]->GetOwner();
500
501 if (!TargetOwner)
502 return AvoidForce;
503
504 // 전방 레이캐스트
505 //FVector2D Forward2D = CurrentData.Velocity.GetSafeNormal();
506
507 //FVector Forward = FVector(Forward2D.X, Forward2D.Y, 0.f);
508
509 float LookAheadDistance = 150.0f + (CurrentData.Velocity.Size() * 0.5f);
510
511 FHitResult Hit;
512 FVector Start = FVector(CurrentData.Position, 10.f);
513
514
515 ForceDir = ForceDir.GetSafeNormal();
516 //FVector End = Start + Forward * LookAheadDistance;
517 FVector End = Start + ForceDir * LookAheadDistance;
518
519 FCollisionQueryParams QueryParams;
520 QueryParams.AddIgnoredActor(Owner);
521
522 if (GetWorld()->LineTraceSingleByObjectType(Hit, Start, End, ECC_WorldStatic, QueryParams))
523 {
524 FVector ObstacleCenter = Hit.Component->Bounds.Origin;
525
526 FVector ImpactToCenter = ObstacleCenter - Hit.ImpactPoint;
527
528 FVector BoidToImpact = Hit.ImpactPoint - FVector(CurrentData.Position, 10.f);
529
530 bool bInsideHit = FVector::DotProduct(ImpactToCenter, BoidToImpact) > 0;
531
532 if (!bInsideHit)
533 {
534 FVector ToMeDir = (Start - ObstacleCenter).GetSafeNormal();
535
536 float AvoidStrength = 1.0f - (Hit.Distance / LookAheadDistance);
537 AvoidForce = FVector2D(ToMeDir * AvoidStrength * Enemies[EnemyIndex]->MaxSpeed);
538 }
539 else
540 {
541 // 장애물 감지 - 측면으로 회피
542 FVector ObstacleNormal = Hit.ImpactNormal;
543 ObstacleNormal.Z = 0; // 수평 회피만
544
545 // 좌우 중 더 나은 방향 선택
546 //FVector Right = FVector::CrossProduct(Forward, FVector::UpVector);
547 FVector Right = FVector::CrossProduct(ForceDir, FVector::UpVector);
548 float RightDot = FVector::DotProduct(Right, ObstacleNormal);
549
550 FVector AvoidDirection = RightDot > 0 ? Right : -Right;
551
552 float AvoidStrength = 1.0f - (Hit.Distance / LookAheadDistance);
553 AvoidForce = FVector2D(AvoidDirection * AvoidStrength * Enemies[EnemyIndex]->MaxSpeed);
554 }
555
556 //길찾기를 안하고있다면 길찾기시작
557 if (Enemies[EnemyIndex]->bPathfinding == false)
558 {
559 FindPath(Enemies[EnemyIndex]);
560 }
561 }
562
563 //DrawDebugLine(GetWorld(), Start, End, FColor::Blue,false, 0.f, 0.f, 5.f);
564 //DrawDebugLine(GetWorld(), Start, Start + FVector(AvoidForce, 10.f), FColor::Yellow,false, 0.f, 0.f, 10.f);
565
566 return AvoidForce;
567}

569FVector2D AFlockingManager::LimitVector(const FVector2D& Vector, float MaxLength)
570{
571 const float LenSq = Vector.SizeSquared();
572 const float MaxLenSq = MaxLength * MaxLength;
573
574 if (LenSq > MaxLenSq)
575 {
576 // 길이가 최대값을 넘으면, 정규화 후 MaxLength로 스케일링
577 return Vector.GetSafeNormal() * MaxLength;
578 }
579
580 // 그렇지 않으면 원래 벡터 반환
581 return Vector;

594void AFlockingManager::DrawSpatialHashGrid(UWorld* World, const FVector2D& Min, const FVector2D& Max, float CellSize,
595 float Z)
596{
597 if (!World) return;
598
599 const int32 StartX = FMath::FloorToInt(Min.X / CellSize);
600 const int32 EndX = FMath::CeilToInt(Max.X / CellSize);
601 const int32 StartY = FMath::FloorToInt(Min.Y / CellSize);
602 const int32 EndY = FMath::CeilToInt(Max.Y / CellSize);
603
604 for (int32 X = StartX; X <= EndX; ++X)
605 {
606 for (int32 Y = StartY; Y <= EndY; ++Y)
607 {
608 FVector Corner1(X * CellSize, Y * CellSize, Z);
609 FVector Corner2((X + 1) * CellSize, Y * CellSize, Z);
610 FVector Corner3((X + 1) * CellSize, (Y + 1) * CellSize, Z);
611 FVector Corner4(X * CellSize, (Y + 1) * CellSize, Z);
612
613 DrawDebugLine(World, Corner1, Corner2, FColor::Green, false, -1.0f, 0, 7.5f);
614 DrawDebugLine(World, Corner2, Corner3, FColor::Green, false, -1.0f, 0, 7.5f);
615 DrawDebugLine(World, Corner3, Corner4, FColor::Green, false, -1.0f, 0, 7.5f);
616 DrawDebugLine(World, Corner4, Corner1, FColor::Green, false, -1.0f, 0, 7.5f);
617 }
618 }
