58void UKPTPlayerAttributeSet::OnRep_MoveSpeed(const FGameplayAttributeData& OldMoveSpeed)
59{
60 GAMEPLAYATTRIBUTE_REPNOTIFY(UKPTPlayerAttributeSet, MoveSpeed, OldMoveSpeed);
61
62 // Get the Character that owns this Attribute Set
63 if (AActor* OwningActor = GetOwningActor())
64 {
65 // Get the movement component
66 if (UCharacterMovementComponent* MovementComp = OwningActor->FindComponentByClass<UCharacterMovementComponent>())
67 {
68 // Set the MaxWalkSpeed to the new MoveSpeed value
69 MovementComp->MaxWalkSpeed = GetMoveSpeed();
70 }
71 }
72}

107void UKPTPlayerAttributeSet::PostGameplayEffectExecute(const FGameplayEffectModCallbackData& Data)
108{
109 Super::PostGameplayEffectExecute(Data);
110
111 if (Data.EvaluatedData.Attribute == GetMoveSpeedAttribute())
112 {
113 // 서버에서만 직접 캐릭터의 이동 속도를 변경합니다.
114 if (GetOwningAbilitySystemComponent()->GetOwnerRole() == ROLE_Authority)
115 {
116 if (AActor* OwningActor = GetOwningActor())
117 {
118 if (UCharacterMovementComponent* MovementComp = OwningActor->FindComponentByClass<UCharacterMovementComponent>())
119 {
120 MovementComp->MaxWalkSpeed = GetMoveSpeed();
121 }
122 }
123 }
