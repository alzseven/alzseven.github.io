9void ULevelSelectItem::LevelDataSetup(const TSoftObjectPtr<UWorld>& LevelAsset)
10{
11 TargetLevel = LevelAsset;
12
13 // IsNull()로 유효하지 않은 포인터는 아닌지 한 번 더 확인합니다.
14 if (!TargetLevel.IsNull())
15 {
16 // ------------------ 여기가 수정된 부분입니다 ------------------
17
18 // GetAssetName() 대신 ToString()을 사용해 경로 문자열을 가져옵니다.
19 // 이 문자열은 에셋이 로드되지 않아도 항상 유효합니다.
20 // 예: "/Game/Maps/MyExampleMap.MyExampleMap"
21 FString LevelPath = TargetLevel.ToString();
22
23 // FPaths::GetBaseFilename은 경로와 확장자를 제거하고 순수한 파일 이름만 남깁니다.
24 // "/Game/Maps/MyExampleMap.MyExampleMap" -> "MyExampleMap"
25 FString LevelDisplayName = FPaths::GetBaseFilename(LevelPath);
26
27 LevelButton->SetMainText(FText::FromString(LevelDisplayName));
28 LevelNameText->SetText(FText::FromString(LevelDisplayName));
29 // -----------------------------------------------------------------
30 }
31
32 else if (LevelButton)
33 {
34 // 만약의 경우를 대비해 비어있다고 표시
35 LevelNameText->SetText(FText::FromString(TEXT("Invalid Level")));
36 LevelButton->SetMainText(FText::FromString(TEXT("Invalid Level")));
37
38 }
39}
